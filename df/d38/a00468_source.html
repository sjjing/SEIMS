67 int IUH_OL::Execute() {
68  this->CheckInputData();
69  this->initialOutputs();
70  // delete value of last time step
71 #pragma omp parallel for
72  for (int n = 0; n < m_nSubbasins + 1; n++) {
73  m_Q_SBOF[n] = 0.f;
74  }
75  //int nt = 0;
76  //float qs_cell = 0.f;
77 
78 #pragma omp parallel for
79  for (int i = 0; i < m_nCells; i++) {
80  //forward one time step
81  for (int j = 0; j < m_cellFlowCols; j++) {
82  if (j != m_cellFlowCols - 1) {
83  m_cellFlow[i][j] = m_cellFlow[i][j + 1];
84  } else {
85  m_cellFlow[i][j] = 0.f;
86  }
87  }
88 
89  float v_rs = m_rs[i];
90  if (v_rs > 0.f) {
91  int min = int(this->m_iuhCell[i][0]);
92  int max = int(this->m_iuhCell[i][1]);
93  int col = 2;
94  for (int k = min; k <= max; k++) {
95  m_cellFlow[i][k] += v_rs / 1000.f * m_iuhCell[i][col] * m_cellArea / m_TimeStep;
96  col++;
97  }
98  }
99  }
100 #pragma omp parallel for
101  for (int i = 0; i < m_nCells; i++) {
102  //add today's flow
103  int subi = (int) m_subbasin[i];
104  if (m_nSubbasins == 1) {
105  subi = 1;
106  } else if (subi >= m_nSubbasins + 1) {
107  throw ModelException(MID_IUH_OL, "Execute", "The subbasin " + ValueToString(subi) + " is invalid.");
108  }
109 
110  m_Q_SBOF[subi] += m_cellFlow[i][0]; //get new value
111 
112  m_OL_Flow[i] = m_cellFlow[i][0];
113  m_OL_Flow[i] = m_OL_Flow[i] * m_TimeStep * 1000.f / m_cellArea; // m3/s -> mm
114  //if(i == 1000) cout << m_OL_Flow[i] << endl;
115  }
116 
117  float tmp = 0.f;
118 #pragma omp parallel for reduction(+:tmp)
119  for (int n = 1; n < m_nSubbasins + 1; n++) {
120  tmp += m_Q_SBOF[n]; //get overland flow routing for entire watershed.
121  }
122  m_Q_SBOF[0] = tmp;
123  return 0;
124 }

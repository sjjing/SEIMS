100 int SNO_SP::Execute() {
101  this->CheckInputData();
102  this->initialOutputs();
105  if (m_snowCoverCoef1 == NODATA_VALUE || m_snowCoverCoef2 == NODATA_VALUE) {
106  PGCommon::getScurveShapeParameter(0.5f, 0.95f, m_snowCover50, 0.95f, &m_snowCoverCoef1, &m_snowCoverCoef2);
107  }
108  // if (this->m_lastSWE == NODATA_VALUE) /// moved to initialOutputs()
109  //this->m_lastSWE = this->m_swe;
110  //If all cells have no snow, snow melt is set to zero.
111  //if (m_swe < 0.01)
112  //{
113  // if (this->m_lastSWE >= 0.01)
114  // {
115  // for (int rw = 0; rw < this->m_nCells; rw++)
116  // m_SM[rw] = 0.f;
117  // }
118  // this->m_lastSWE = this->m_swe;
119 
120  // return 0;
121  //}
122 
124  // which only need to computed once.
125  int d = JulianDay(this->m_date);
126  float sinv = float(sin(2.f * PI / 365.f * (d - 81.f)));
127  float cmelt = (m_csnow6 + m_csnow12) / 2.f + (m_csnow6 - m_csnow12) / 2.f * sinv;
128 #pragma omp parallel for
129  for (int rw = 0; rw < m_nCells; rw++) {
131  m_packT[rw] = m_packT[rw] * (1 - m_lagSnow) + m_tMean[rw] * m_lagSnow;
133  m_SA[rw] = m_SA[rw] + m_SR[rw] - m_SE[rw];
134  if (m_tMean[rw] < m_tsnow)
135  {
136  m_SA[rw] += m_kblow * m_Pnet[rw];
137  m_Pnet[rw] *= (1.f - m_kblow);
138  }
139 
140  if (m_SA[rw] < 0.01) {
141  m_SM[rw] = 0.f;
142  } else {
143  float dt = m_tMax[rw] - m_t0;
144  if (dt < 0) {
145  m_SM[rw] = 0.f; //if temperature is lower than t0, the snowmelt is 0.
146  } else {
147  //calculate using eq. 1:2.5.2 SWAT p58
148  m_SM[rw] = cmelt * ((m_packT[rw] + m_tMax[rw]) / 2.f - m_t0);
149  // adjust for areal extent of snow cover
150  float snowCoverFrac = 0.f; //fraction of HRU area covered with snow
151  if (m_SA[rw] < m_snowCoverMax) {
152  float xx = m_SA[rw] / m_snowCoverMax;
153  snowCoverFrac = xx / (xx + exp(m_snowCoverCoef1 = m_snowCoverCoef2 * xx));
154  } else {
155  snowCoverFrac = 1.f;
156  }
157  m_SM[rw] *= snowCoverFrac;
158  if (m_SM[rw] < 0.f) m_SM[rw] = 0.f;
159  if (m_SM[rw] > m_SA[rw]) m_SM[rw] = m_SA[rw];
160  m_SA[rw] -= m_SM[rw];
161  m_Pnet[rw] += m_SM[rw];
162  if (m_Pnet[rw] < 0.f) m_Pnet[rw] = 0.f;
163  }
164  }
165  }
166  //this->m_lastSWE = this->m_swe;
167  return 0;
168 }

65 int SNO_DD::Execute() {
66  this->CheckInputData();
67 
68  this->initialOutputs();
69 
70 #pragma omp parallel for
71  for (int i = 0; i < m_nCells; i++) {
72  //float se = 0.f;
73  //float sr = 0.f;
74  //if (m_SE != NULL)
75  // se = m_SE[i];
76  //if (m_SR != NULL)
77  // sr = m_SR[i];
78 
79  m_SA[i] = m_SA[i] + m_SR[i] - m_SE[i];
80 
81  if (FloatEqual(m_SA[i], 0.f) && FloatEqual(m_Pnet[i], 0.f)) {
82  m_SM[i] = 0.f;
83  continue;
84  }
85 
86  if (m_tMean[i] < m_tsnow)
87  {
88  m_SM[i] = 0.f;
89  m_SA[i] += m_kblow * m_Pnet[i];
90  m_Pnet[i] *= (1.f - m_kblow);
91  }
92  /*else if (m_tMean[i] > m_tsnow && m_tMean[i] <= m_t0)
93  {
94  m_SM[i] = 0.f;
98  }*/
99  else {
100  float dt = (m_tMean[i] - m_t0);
101  if (dt < 0.f) {
102  m_SM[i] = 0.f; //if temperature is lower than t0, the snowmelt is 0.
103  } else {
104  m_SM[i] = m_csnow * dt + m_crain * m_Pnet[i] * dt;
105  if (m_SM[i] < 0.f) m_SM[i] = 0.f;
106  if (m_SM[i] > m_SA[i]) m_SM[i] = m_SA[i];
107  m_SA[i] -= m_SM[i];
108  m_Pnet[i] += m_SM[i];
109  if (m_Pnet[i] < 0.f) m_Pnet[i] = 0.f;
110  }
111  }
112  }
113  return 0;
114 }

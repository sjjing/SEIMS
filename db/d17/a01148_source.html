47 void Muskingum::GetCofficients(float reachLength, float waterDepth, float s0, float v0, MuskWeights &weights) {

49  //float x = 0.5f * (1 - waterDepth/(m_beta*reachLength*s0));
50  //if (x < 0)
51  // x = 0.01f;
52  float x = m_msk_x;
53  //float v = sqrt(s0) * pow(waterDepth, TWO_THIRDS) / manning_n;

67  //get coefficient

69  weights.c1 = (delta_t - min) / temp;
70  weights.c2 = (delta_t + min) / temp;
71  weights.c3 = (max - delta_t) / temp;
72  weights.c4 = 2 * delta_t / temp;
73 
74  //make sure any coefficient is positive
75  if (weights.c1 < 0) {
76  weights.c2 += weights.c1;
77  weights.c1 = 0.0f;
78  }

227 bool Muskingum::ChannelFlow(int iReach, int iCell, int id, float qgEachCell) {
228  float qUpNew = 0.f;
229 
230  if (iReach == 0 && iCell == 0) {
231  qUpNew = m_qUpReach;
232  }
233 
234  // inflow from upstream channel
235  if (iCell == 0)// inflow of this cell is the last cell of the upstream reach
236  {
237  for (size_t i = 0; i < m_reachUpStream[iReach].size(); ++i) {
238  int upReachId = m_reachUpStream[iReach][i];
239  if (upReachId >= 0) {
240  int upCellsNum = m_reachs[upReachId].size();
241  int upCellId = m_reachs[upReachId][upCellsNum - 1];
242  qUpNew += m_qCh[upReachId][upCellsNum - 1];
243  }
244  }
245  } else {
246  qUpNew = m_qCh[iReach][iCell - 1];
247  }
248 
249  float dx = m_flowLen[iReach][iCell];
250  float area = dx * m_chWidth[id];
251 
252  // lateral flow
253  float qLat = m_prec[id] / 1000.f * area / m_dt;
254 
255  if (m_qs != NULL) {
256  qLat += m_qs[id];
257  }
258 
259  if (m_qi != NULL) {
260  qLat += m_qi[id];
261  }
262 
263  qLat += qgEachCell;
264 
265  //if (iReach == 364)
266  // cout << qLat << qUpNew << endl;
267  qUpNew += qLat;
268 
269  float waterDepth = m_chStorage[iReach][iCell] / area;
270 
271  MuskWeights weights;
272  GetCofficients(dx, waterDepth, m_s0[id], m_v0[iReach], weights);
273  int stepCount = int(m_dt / weights.dt);
274  //start the loop for every delta t
275  float qNew = 0.f;
276 
277  float q = 0.f;
278  for (int i = 0; i < stepCount; i++) {
279 
280  //GetCofficients(dx, waterDepth, m_s0[i], m_v0[iReach], weights);
281  // unit discharge
282  qNew = weights.c1 * qUpNew + weights.c2 * m_qUpCh[iReach][iCell] + weights.c3 * qNew;
283 
284  if (qNew != qNew) {
285  cout << "Error in function Muskingum::ChannelFlow. \n";
286  cout << "The weights are: " << weights.c1 << ", " << weights.c2 << ", " << weights.c3 << endl;
287  cout << "qUpNew: " << qUpNew << "\tqUpPre: " << m_qUpCh[iReach][iCell] << "\tqNew: " << qNew << endl;
288  //throw ModelException(MID_CH_MSK, "ChannelFlow", "Error occurred.");
289  return false;
290  }
291 
292  float tmp = m_chStorage[iReach][iCell] + (qUpNew - qNew) * weights.dt;
293  if (tmp > 0.f) {
294  m_chStorage[iReach][iCell] = tmp;
295  m_qCh[iReach][iCell] = qNew;
296  } else {
297  m_qCh[iReach][iCell] = m_chStorage[iReach][iCell] / weights.dt;
298  m_chStorage[iReach][iCell] = 0.f;
299  }
300 
301  m_qUpCh[iReach][iCell] = qUpNew;
302  q += m_qCh[iReach][iCell];
303  }
304 
305  q /= stepCount;
306  m_qCh[iReach][iCell] = q;
307  return true;

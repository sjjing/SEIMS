106 int IUH_IF::Execute() {
107  this->CheckInputData();
108 
109  this->initialOutputs();
110 
111 #pragma omp parallel for
112  for (int n = 0; n < m_nsub + 1; n++) {
113  m_Q_SBIF[n] = 0.0f; // delete value of last time step
114  }
115 
116 
117  //int nt = 0;
118  //float qs_cell = 0.0f;
119  float area = m_CellWidth * m_CellWidth;
120 
121  //#pragma omp parallel for
122  for (int i = 0; i < m_nCells; i++) {
123  //forward one time step
124  for (int j = 0; j < m_cellFlowCols; j++) {
125  if (j != m_cellFlowCols - 1) {
126  m_cellFlow[i][j] = m_cellFlow[i][j + 1];
127  } else {
128  m_cellFlow[i][j] = 0.0f;
129  }
130  }
131 
132  //add today's flow
133  int subi = (int) m_subbasin[i];
134 
135  if (m_nsub == 1) {
136  subi = 1;
137  } else if (subi >= m_nsub + 1) {
138  throw ModelException("IUH_IF", "Execute", "The subbasin " + ValueToString(subi) + " is invalid.");
139  }
140 
141  float v_rs = m_ssru[i];
142  if (v_rs > 0.f) {
143  int min = int(this->m_iuhCell[i][0]);
144  int max = int(this->m_iuhCell[i][1]);
145  int col = 2;
146  for (int k = min; k <= max; k++) {
147  this->m_cellFlow[i][k] += v_rs / 1000.0f * m_iuhCell[i][col] * area / m_TimeStep;
148  col++;
149  }
150  }
151  //#pragma omp critical
152  {
153  m_Q_SBIF[subi] += this->m_cellFlow[i][0]; //get new value
154  }
155  }
156 
157  float tmp = 0.f;
158  //#pragma omp parallel for reduction(+:tmp)
159  for (int n = 1; n < m_nsub + 1; n++) {
160  tmp += m_Q_SBIF[n]; //get overland flow routing for entire watershed.
161  }
162  m_Q_SBIF[0] = tmp;
163 
164  return 0;

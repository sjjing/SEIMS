60 int HS_WB::Execute() {
61 
62  initialOutputs();
63 
65 #pragma omp parallel for
66  for (int i = 1; i <= m_nReaches; i++) {
67  m_qsTemp[i] = 0.f;
68  m_qiTemp[i] = 0.f;
69  }
70 
71  int reachId;
72  for (int i = 0; i < m_nCells; i++) {
73  if (m_streamLink[i] > 0) {
74 #ifdef MULTIPLY_REACHES
75  reachId = int(m_streamLink[i]);
76 #else
77  reachId = 1;
78 #endif
79  m_qsTemp[reachId] += m_qs[i];
80  if (m_qi != NULL) {
81  m_qiTemp[reachId] += m_qi[i];
82  }
83  }
84  }
85 
86  m_qsInput[0] = 0.f;
87  m_qiInput[0] = 0.f;
88  for (int i = 1; i <= m_nReaches; i++) {
89  m_qsInput[i] = (m_qsInput[i] * (m_tsCounter - 1) + m_qsTemp[i]) / m_tsCounter;
90  m_qiInput[i] = (m_qiInput[i] * (m_tsCounter - 1) + m_qiTemp[i]) / m_tsCounter;
91 
92  m_qsInput[0] += m_qsInput[i];
93  m_qiInput[0] += m_qiInput[i];
94  }
95 
96  m_tsCounter++;
97 
98  return 0;
99 }

174 int PETPenmanMonteith::Execute() {
175  //check the data
176  if (!this->CheckInputData()) return false;
177  initialOutputs();
178  int d = JulianDay(this->m_date);
179  //do the execute
180 #pragma omp parallel for
181  for (int j = 0; j < m_nCells; ++j) {
183  /* update base zero total heat units, src code from SWAT, subbasin.f
184  if (tmpav(j) > 0. .and. phutot(hru_sub(j)) > 0.01) then
185  phubase(j) = phubase(j) + tmpav(j) / phutot(hru_sub(j))
186  end if*/
187  if (m_jday == 1) m_phuBase[j] = 0.f;
188  if (m_tMean[j] > 0.f && m_phutot[j] > 0.01f) {
189  m_phuBase[j] += m_tMean[j] / m_phutot[j];
190  }
191 
193  // compute net radiation
194  //net short-wave radiation for PET (from swat)
195  float raShortWave = m_sr[j] * (1.f - 0.23f);
196  if (m_tMean[j] < this->m_tSnow) { //if the mean t < snow t, consider the snow depth is larger than 0.5mm.
197  raShortWave = m_sr[j] * (1.f - 0.8f);
198  }
199 
200  //calculate the max solar radiation
201  float srMax = 0.f;
202  MaxSolarRadiation(d, m_cellLat[j], m_dayLen[j], srMax);
203  //calculate net long-wave radiation
204  //net emissivity equation 2.2.20 in SWAT manual
205  float satVaporPressure = SaturationVaporPressure(m_tMean[j]);//kPa
206  float actualVaporPressure = 0.f;
207  if (m_rhd[j] > 1) {
208  actualVaporPressure = m_rhd[j] * satVaporPressure * 0.01f;
209  } else {
210  actualVaporPressure = m_rhd[j] * satVaporPressure;
211  }
212  m_vpd[j] = satVaporPressure - actualVaporPressure;
213  float rbo = -(0.34f - 0.139f * sqrt(actualVaporPressure)); //P37 1:1.2.22
214  //cloud cover factor equation 2.2.19
215  float rto = 0.0f;
216  if (srMax >= 1.0e-4f) {
217  rto = 0.9f * (m_sr[j] / srMax) + 0.1f;
218  }
219  //net long-wave radiation equation 2.2.21
220  float tk = m_tMean[j] + 273.15f;
221  float raLongWave = rbo * rto * 4.9e-9f * pow(tk, 4.f);
222  // calculate net radiation
223  float raNet = raShortWave + raLongWave;
224 
226  //calculate the slope of the saturation vapor pressure curve
227  float dlt = 4098.f * satVaporPressure / pow((m_tMean[j] + 237.3f), 2.f);
228 
229  //calculate latent heat of vaporization(MJ/kg, from swat)
230  float latentHeat = 2.501f - 0.002361f * m_tMean[j];
231 
232  //calculate mean barometric pressure(kPa)
233  float pb = 101.3f - m_elev[j] * (0.01152f - 0.544e-6f * m_elev[j]); //p53 1:2.3.8 P
234  //psychrometric constant(kPa/deg C)
235  float gma = 1.013e-3f * pb / (0.622f * latentHeat);//p53 1:2.3.7
236 
237  float rho = 1710.f - 6.85f * m_tMean[j];//p127 2:2.2.19
238  //aerodynamic resistance to sensible heat and vapor transfer(s/m)
239  float rv = 114.f / (m_ws[j] * pow(170.f / 1000.f, 0.2f)); // P127 2:2.2.20 ??the wind speed need modified?
240  //canopy resistance(s/m)
241  float rc = 49.f / (1.4f - 0.4f * m_co2 / 330.f); //P128 2:2.2.22
242  float petValue = (dlt * raNet + gma * rho * m_vpd[j] / rv) /
243  (latentHeat * (dlt + gma * (1.f + rc / rv))); //P122 2:2.2.2
244  petValue = m_petFactor * max(0.f, petValue);
245  m_pet[j] = petValue;
246  //*********************************************************
247  //The albedo would be obtained from plant growth module. But now it is assumed to be a constant.
248  //After the plant growth module is completed, the following codes should be removed.
249  //float albedo = 0.8f;
250  //if(m_tMean[j] > this->m_tSnow) albedo = 0.23f;
252  //*********************************************************
253  // calculate net short-wave radiation for max plant PET
254  float raShortWavePlant = m_sr[j] * (1.0f - m_albedo[j]);
255  float raNetPlant = raShortWavePlant + raLongWave;
256 
257  float epMax = 0.f;
258  if (m_growCode[j] > 0) //land cover growing
259  {
260  //determine wind speed and height of wind speed measurement
261  //adjust to 100 cm (1 m) above canopy if necessary
262  float zz = 0.f; //height at which wind speed is determined(cm)
263  if (m_cht[j] <= 1.f) {
264  zz = 170.f;
265  } else {
266  zz = m_cht[j] * 100.f + 100.f;
267  }
268  //wind speed at height zz(m/s)
269  float uzz = m_ws[j] * pow(zz / 1000.f, 0.2f);
270 
271  //calculate canopy height in cm
272  float chz = 0.f;
273  if (m_cht[j] < 0.01f) {
274  chz = 1.0f;
275  } else {
276  chz = m_cht[j] * 100.f;
277  }
278 
279  //calculate roughness length for momentum transfer
280  float zom = 0.f;
281  if (chz <= 200.f) {
282  zom = 0.123f * chz;
283  } else {
284  zom = 0.058f * pow(chz, 1.19f);
285  }
286 
287  //calculate roughness length for vapor transfer
288  float zov = 0.1f * zom;
289 
290  //calculate zero-plane displacement of wind profile
291  float d = 0.667f * chz;
292 
293  //calculate aerodynamic resistance
294  float rv = log((zz - d) / zom) * log((zz - d) / zov);
295  rv = rv / (pow(0.41f, 2.f) * uzz);
296 
297  //adjust stomatal conductivity for low vapor pressure
298  //this adjustment will lower maximum plant ET for plants
299  //sensitive to very low vapor pressure
300  float xx = m_vpd[j] - 1.f;//difference between vpd and vpthreshold
301  //amount of vapor pressure deficit over threshold value
302  float fvpd = 1.f;
303  if (xx > 0.f) {
304  fvpd = max(0.1f, 1.f - m_vpd2[j] * xx);
305  }
306  float gsi_adj = m_gsi[j] * fvpd;
307 
308  if (gsi_adj > 1.e-6f) {
309  //calculate canopy resistance
310  float rc = 1.f / gsi_adj; //single leaf resistance
311  rc = rc / (0.5f * (m_lai[j] + 0.01f) * (1.4f - 0.4f * m_co2 / 330.f));
312  //calculate maximum plant ET
313  epMax = (dlt * raNetPlant + gma * rho * m_vpd[j] / rv) / (latentHeat * (dlt + gma * (1.f + rc / rv)));
314  epMax = max(0.f, epMax);
315  epMax = min(epMax, petValue);
316  } else {
317  epMax = 0.f;
318  }
319  }
320  m_ppt[j] = epMax;
321  }
322  return 0;
323 }

193 int SUR_GreenAmpt::Execute() {
194  this->CheckInputData();
195 
196  this->initalOutputs();
197 
198  // define local variables
199  float sol_por, dthet, SW;
200  float adj_hc, sol_k, cnday, psidt, tst, f1, pNet;
201  float rateinf0, rateinf, cuminf, rintns, wfmp;
202 
203 #pragma omp parallel for
204  for (int iCell = 0; iCell < m_cellSize; iCell++) {
205  //initialize the variables
206  rateinf = 0.0f;
207  cuminf = 0.0f;
208  pNet = 0.0f;
209  float pcp = 0.0f; //rainfall - interception
210  float dep = 0.0f; //depression storage
211  float sna = 0.0f; // snow accumulation
212  float snm = 0.0f; //snowmelt
213  float Tsoil = 0.0f; //soil temperature
214  float t = 0.0f; // air temperature
215  float sm_frozen = 0.0f; // frozen soil moisture relative to saturation
216  float soilmoist = 0.0f; //soil moisture
217  //set values to variables
218  pcp = m_P_NET[iCell];
219  //if (m_SD == NULL) // the depression storage module is not available, set the initial depression storage
220  //{
221  // dep = m_Depre_in * m_Depression[iCell];
222  //}
223  //else
224  //{
225  dep = m_SD[iCell];
226  //}
227  sna = m_SA[iCell];
228  snm = m_SM[iCell];
229  t = (m_tMin[iCell] + m_tMax[iCell]) / 2;
230  Tsoil = m_TS[iCell];
231  sm_frozen = m_Sfrozen * m_porosity[iCell];
232  soilmoist = m_soilMoisture[iCell];
233  //account for the effects of snowmelt and soil temperature
234  // snow, without snow melt
235  if (t <= m_Tsnow) {
236  pNet = 0.0f;
237  }
238  // rain on snow, no snow melt
239  else if (t > m_Tsnow && t <= m_T0 && m_SA[iCell] > pNet) {
240  pNet = 0.0f;
241  } else {
242  pNet = pcp + dep + snm; // default
243  }
244 
245  //
246  if (pNet > 0.0f) {
247  //for frozen soil
248  if (Tsoil <= m_Tsoil && soilmoist >= sm_frozen) {
249  m_PE[iCell] = pNet;
250  m_INFIL[iCell] = 0.0f;
251  }
252  //for GA method
253  else {
254  //calculate CN
255  cnday = 0.0f;
256  cnday = Calculate_CN(iCell);
257  // calculate effective hydraulic conductivity
258  sol_k = m_Conductivity[iCell];
259  adj_hc = (56.82f * pow(sol_k, 0.286f)) / (1.f + 0.051f * exp(0.062f * cnday)) - 2.0f;
260  if (adj_hc <= 0.0) {
261  adj_hc = 0.001f;
262  }
263  // calculate delta theta: the change in volumetric moisture content across the wetting front(mm/mm)
264  dthet = 0.0f;
265  float soilw = 0.0f;
266  //set infiltration rate for last time step or the initial infiltration rate
267  rateinf0 = m_INFRate[iCell];
268 
269  SW = m_soilMoisture[iCell] * m_rootDepth[iCell];
270  float fieldcap = m_fieldCap[iCell] * m_rootDepth[iCell];
271  if (SW >= fieldcap) {
272  soilw = 0.999f * fieldcap;
273  } else {
274  soilw = SW;
275  }
276  sol_por = m_porosity[iCell];
277  dthet = (1.0f - soilw / fieldcap) * sol_por * 0.95f;
278 
279  // calculate wetting front matric potential
280  psidt = 0.0;
281  //sol_clay = m_clay[iCell] / 100;
282  //sand = m_sand[iCell] / 100;
283  wfmp = this->m_wfmp[iCell];
284  //wfmp = Calculate_WFMP(sol_por,sol_clay,sand);
285  psidt = dthet * wfmp;
286 
287  //calculate rainfall intensity for time step
288  rintns = 60.0f * pNet / m_TimeStep;
289  //if rainfall intensity is less than infiltration rate
290  //everything will infiltrate
291  cuminf = 0.0f;
292  if (rateinf0 >= rintns) {
293  cuminf = rintns * m_TimeStep / 60.0f;
294  } else {
295  //if rainfall intensity is greater than infiltration rate
296  //find cumulative infiltration for time step by successive
297  //substitution
298  tst = 0.0f;
299  tst = adj_hc * m_TimeStep / 60.0f;
300  do {
301  f1 = 0.0f;
302  f1 = cuminf + adj_hc * m_TimeStep / 60.0f + psidt * log((tst + psidt) / (cuminf + psidt));
303  if (abs(f1 - tst) <= 0.001) {
304  cuminf = f1;
305  break;
306  } else {
307  tst = 0.0f;
308  tst = f1;
309  }
310  } while (1);
311  }
312  //calculate new rate of infiltration
313  if (cuminf != 0) {
314  rateinf = adj_hc * (psidt / (cuminf + 0.000001f) + 1.0f);
315  }
316  m_INFRate[iCell] = rateinf; //set new rate of infiltration
317  m_INFIL[iCell] = cuminf;
318  m_PE[iCell] = pcp + snm - cuminf;
319  }
320  } else {
321  m_PE[iCell] = 0.0f;
322  m_INFIL[iCell] = 0.0f;
323  }
324  // check the output data
325  if (m_INFIL[iCell] > 10000.0f || m_INFIL[iCell] < 0.0f) {
326  //string datestr = getDate(&m_date);
327  ostringstream oss;
328  oss << "Cell(Row:" << m_mask[iCell][0] << ", Col:" << m_mask[iCell][1] << "\n Infiltration =" <<
329  m_INFIL[iCell]
330  << "\n Precipitation(mm) = " << pNet << "\n InfiltrationRate(m/h) = " << rateinf << "\n";
331  throw ModelException("SUR_GreenAmpt", "Execute",
332  "Output data error: infiltration is out of reasonable range. Where:\n"
333  + oss.str() + "Please contact the module developer. ");
334  }
335  }
336  return 0;

453 float SUR_GreenAmpt::Calculate_CN(int cell) {
454  float sw, s, CNday, xx;
455 
456  s = 0.;
457  sw = this->m_soilMoisture[cell] * this->m_rootDepth[cell];
458  xx = this->m_w1[cell] - this->m_w2[cell] * sw;
459  if (xx < -20.f) {
460  xx = -20.;
461  }
462  if (xx > 20.f) {
463  xx = 20.f;
464  }
465  // traditional CN method (function of soil water)
466  if ((sw + exp(xx)) > 0.001f) {
467  s = this->m_sMax[cell] * (1.f - sw / (sw + exp(xx))); //2:1.1.6
468  }
469 
470  CNday = 25400.f / (s + 254.f); //2:1.1.11
471  return CNday;

474 void SUR_GreenAmpt::initalW1W2() {
475  this->m_w1 = new float[this->m_cellSize];
476  this->m_w2 = new float[this->m_cellSize];
477  this->m_sMax = new float[this->m_cellSize];
478 
479  for (int i = 0; i < this->m_cellSize; i++) {
480  float cnn = this->m_CN2[i];
481  float fieldcap = this->m_fieldCap[i] * this->m_rootDepth[i];
482  float wsat = this->m_porosity[i] * this->m_rootDepth[i];
483  float c1, c3, c2, smx, s3, rto3, rtos, xx, wrt1, wrt2;
484  c2 = 100.0f - cnn;
485  c1 = cnn - 20.f * c2 / (c2 + exp(2.533f - 0.0636f * c2)); //CN1 2:1.1.4
486  c1 = max(c1, 0.4f * cnn);
487  c3 = cnn * exp(0.006729f * c2); //CN3 2:1.1.5
488 
489  //calculate maximum retention parameter value
490  smx = 254.f * (100.f / c1 - 1.f); //2:1.1.2
491 
492  //calculate retention parameter value for CN3
493  s3 = 254.f * (100.f / c3 - 1.f);
494 
495  rto3 = 1.f - s3 / smx;
496  rtos = 1.f - 2.54f / smx;
497 
498  xx = log(fieldcap / rto3 - fieldcap);
499  wrt2 = (xx - log(wsat / rtos - wsat)) /
500  (wsat - fieldcap); //soilWater :completely saturated (= soil_por - sol_wp) w1 2:1.1.8
501  wrt1 = xx + (fieldcap * wrt2); //w2 2:1.1.7
502 
503  this->m_w1[i] = wrt1;
504  this->m_w2[i] = wrt2;
505  this->m_sMax[i] = smx;
506  }

521 float SUR_GreenAmpt::Calculate_WFMP(float sol_por, float sol_clay,
522  float sand) //this function calculated the wetting front matric potential
523 {
524  float wfmp = 10.0f * exp(6.5309f - 7.32561f * sol_por + 3.809479f * pow(sol_por, 2) + 0.001583f *
525  pow(sol_clay, 2) +
526  0.000344f * sand * sol_clay - 0.049837f * sol_por * sand + 0.001608f *
527  pow(sol_por, 2) * pow(sand, 2) +
528  0.001602f * pow(sol_por, 2) * pow(sol_clay, 2) -
529  0.0000136f * pow(sand, 2) * sol_clay - 0.003479f * pow(sol_clay, 2) *
530  sol_por - 0.000799f * pow(sand, 2) * sol_por);

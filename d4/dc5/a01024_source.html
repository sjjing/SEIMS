54  dic = dict()
55  for j in range(len(site_data_items[i])):
56  if StringClass.string_match(site_flds[j], StationFields.id):
57  dic[StationFields.id] = int(site_data_items[i][j])
58  elif StringClass.string_match(site_flds[j], StationFields.name):
59  # unicode(site_data_items[i][j], 'gb2312')
60  dic[StationFields.name] = site_data_items[i][j]
61  elif StringClass.string_match(site_flds[j], StationFields.x):
62  dic[StationFields.x] = float(site_data_items[i][j])
63  elif StringClass.string_match(site_flds[j], StationFields.y):
64  dic[StationFields.y] = float(site_data_items[i][j])
65  elif StringClass.string_match(site_flds[j], StationFields.lon):
66  dic[StationFields.lon] = float(site_data_items[i][j])
67  elif StringClass.string_match(site_flds[j], StationFields.lat):
68  dic[StationFields.lat] = float(site_data_items[i][j])
69  elif StringClass.string_match(site_flds[j], StationFields.elev):
70  dic[StationFields.elev] = float(site_data_items[i][j])
71  elif StringClass.string_match(site_flds[j], StationFields.outlet):
72  dic[StationFields.outlet] = float(site_data_items[i][j])
73  dic[StationFields.type] = site_type
74  curfilter = {StationFields.id: dic[StationFields.id],
75  StationFields.type: dic[StationFields.type]}
76  hydro_clim_db[DBTableNames.sites].find_one_and_replace(curfilter, dic, upsert=True)
77 
78  if dic[StationFields.id] not in sites_loc.keys():
79  sites_loc[dic[StationFields.id]] = SiteInfo(dic[StationFields.id],
80  dic[StationFields.name],
81  dic[StationFields.lat],
82  dic[StationFields.lon],
83  dic[StationFields.x],
84  dic[StationFields.y],
85  dic[StationFields.elev])
86  hydro_clim_db[DBTableNames.sites].create_index([(StationFields.id, ASCENDING),
87  (StationFields.type, ASCENDING)])
88  return sites_loc

109  from shapely.wkt import loads as shapely_loads
110  shapely_objects = []
111  id_list = []
112  # print input_shape

116  feat = lyr.GetFeature(n)

118  shapely_objects.append(wkt_feat)
119  if isinstance(id_field, unicode):
120  id_field = id_field.encode()
121  id_index = feat.GetFieldIndex(id_field)
122  id_list.append(feat.GetField(id_index))
123  return shapely_objects, id_list

174  """Workflow"""

176  thiessen_file_list = [cfg.meteo_sites_thiessen, cfg.prec_sites_thiessen]

184  FieldNames.subbasin_id, thiessen_file_list,
185  cfg.thiessen_field, type_list,
186  cfg.storm_mode, cfg.cluster)
187 
188  # 2. Import geographic information of each sites to Hydro-Climate database
189  c_list = clim_db.collection_names()
190  tables = [DBTableNames.sites, DBTableNames.var_desc]
191  for tb in tables:
192  if not StringClass.string_in_list(tb, c_list):
193  clim_db.create_collection(tb)
194  ImportHydroClimateSites.variable_table(clim_db, cfg.hydro_climate_vars)
195  site_m_loc = ImportHydroClimateSites.sites_table(clim_db, cfg.Meteo_sites, DataType.m)
196  site_p_loc = ImportHydroClimateSites.sites_table(clim_db, cfg.prec_sites, DataType.p)

187  //set the input data which was not available right now, this will be delete when the data or module is available
188  //setNotAvailableInput();
189 
190  CheckInputData();
191 
192  initialOutputs();
193  //StatusMsg("executing SplashEro_Park");
194 #pragma omp parallel for
195  for (int i = 0; i < m_nCells; i++) {
196  //intensity in mm/h
197  float RainInten = m_Rain[i] * 3600.f / m_TimeStep;
198 
199  //rainfall between plants in mm //Rainc->Drc = Rain->Drc * _dx/DX->Drc;
200  // correction for slope dx/DX, water spreads out over larger area
201  float s = max(0.001f, m_Slope[i]);
202  float S0 = sin(atan(s));
203  float Dm = 0.00124f * pow(RainInten, 0.182f);
204  float waterdepth = (m_sr[i] + m_depression[i]) / 1000.f; // mm convert to m
205  float Fw, Dr;
206  if (waterdepth > Dm) {
207  Fw = exp(1.f - waterdepth / Dm);
208  } else {
209  Fw = 1.f;
210  }
211  // kg/(m2*min)
212  Dr = m_Omega * Fw * m_USLE_C[i] * m_USLE_K[i] * pow(RainInten, 2.f) * (2.96f * pow(S0, 0.79f) + 0.56f);
213  // convert kg/(m2*min) to kg/cell
214  float cellareas = (m_CellWith / cos(atan(s))) * m_CellWith;
215  m_DETSplash[i] = Dr * (m_TimeStep / 60.f) * cellareas;
216  // // Deal with all exceptions:
217  //if (m_SnowCover[i]>0)
218  //{
219  // m_DETSplash[i] = (1-m_SnowCover[i]) * m_DETSplash[i]; //no splash on snow deck
220  //}
221 
222  }
223  //StatusMsg("end of executing SplashEro_Park");
224  return 0;

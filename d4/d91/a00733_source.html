334 void KinWavSed_OL::GetTransportCapacity(int id) {
335  float q, S0, K;
336  q = m_V[id] * m_WH[id] *
337  60.f; // m2/s -> m2/min // m_Qkin[id]*60; // convert to m3/min
338  float s = max(0.001f, m_Slope[id]);
339  S0 = sin(atan(s));
340  K = m_USLE_K[id];
341  float threadhold = 0.046f;
342  if (q <= 0.f) {
343  m_Ctrans[id] = 0.f;
344  } else {
345  if (q < threadhold) {
346  m_Ctrans[id] = m_eco1 * K * S0 * sqrt(q); // kg/(m*min) kg per unit width per minute
347  } else {
348  m_Ctrans[id] = m_eco2 * K * S0 * pow(q, 2.0f);
349  }
350  // kg/(m*min) -> convert to kg/m3
351  m_Ctrans[id] = m_Ctrans[id] / q;
352  }

355 void KinWavSed_OL::GetSedimentInFlow(int id) {
356  float TC, Df, Dsp, Deposition, concentration, Vol;
357  GetTransportCapacity(id);
358  TC = m_Ctrans[id]; //kg/m3
359  CalcuFlowDetachment(id);
360  Df = m_DETOverland[id]; //kg/cell
361  Dsp = m_DETSplash[id]; //kg/cell
362 
363  m_Sed_kg[id] += (Df + Dsp); //sediment in flow
364  //while the sediment in flow is more than the capacity of the overland flow,
365  // then the sediment deposition would happen; if the velocity of flow is reduced, the TC is decreased correspondingly,
366  // then, the sediment deposition may take place.
367  //MaxConcentration(m_Vol[id], m_Sed_kg[id], id);
368  Vol = m_Vol[id];
369  if (Vol > 0) {
370  concentration = m_Sed_kg[id] / Vol; // convert to kg/m3
371  } else {
372  concentration = 0;
373  }
374  Deposition = max(concentration - TC, 0.0f); //kg/m3, >0
375  if (Deposition > 0) {
376  m_Sed_kg[id] = TC * Vol;
377  }
378  m_SedDep[id] = Deposition * Vol; //kg/cell
379 }

443 void KinWavSed_OL::CalcuFlowDetachment(int i) //i is the id of the cell in the grid map
444 {
445  // correction for slope dx/DX, water spreads out over larger area
446  float s = max(0.001f, m_Slope[i]);
447  float S0 = sin(atan(s));
448  float waterdepth = m_WH[i] / 1000.f; // mm convert to m
449  //float waterdepth = m_Runoff[i] / 1000; // mm convert to m
450 
451  //using Foster equation to calculate overland flow detachment
452  float Df, waterden, g, shearStr; //,q
453  waterden = 1000;
454  g = 9.8f;
455  shearStr = waterden * g * waterdepth * S0;
456  Df = m_Ccoe * m_USLE_C[i] * m_USLE_K[i] * pow(shearStr, 1.5f);
457  /*q = m_Q[i];
458  Df = m_Ccoe * m_USLE_C[i] * m_USLE_K[i] * q * S0;*/
459  // convert kg/(m2*min) to kg/cell
460  float cellareas = (m_CellWidth / cos(atan(s))) * m_CellWidth;
461  m_DETOverland[i] = Df * (m_TimeStep / 60) * cellareas;
462 }

495 float KinWavSed_OL::complexSedCalc(float Qj1i1, float Qj1i, float Qji1, float Sj1i, float Sji1, float alpha, float dt,
496  float dx) {
497  float Sj1i1, Cavg, Qavg, aQb, abQb_1, A, B, C, s = 0.f;
498  const float beta = 0.6f;
499 
500  if (Qj1i1 < 1e-6) {
501  return (0);
502  }
503 
504  Qavg = 0.5f * (Qji1 + Qj1i);
505  if (Qavg <= 1e-6) {
506  return (0);
507  }
508 
509  Cavg = (Sj1i + Sji1) / (Qj1i + Qji1);
510  aQb = alpha * pow(Qavg, beta);
511  abQb_1 = alpha * beta * pow(Qavg, beta - 1);
512 
513  A = dt * Sj1i;
514  B = -dx * Cavg * abQb_1 * (Qj1i1 - Qji1);
515  C = (Qji1 <= 1e-6 ? 0 : dx * aQb * Sji1 / Qji1);
516  if (Qj1i1 > 1e-6) {
517  Sj1i1 = (dx * dt * s + A + C + B) / (dt + dx * aQb / Qj1i1); //why s = 0 ?
518  } else {
519  Sj1i1 = 0;
520  }
521  Sj1i1 = max(0.f, Sj1i1);
522  return Sj1i1;
523 }

525 void KinWavSed_OL::OverlandflowSedRouting(int id) {
526  //sum the sediment of the upstream overland flow
527  float flowwidth = m_FlowWidth[id];
528  float Sin = 0.0f;
529  float Qin = 0.0f;
530  for (int k = 1; k <= (int) m_flowInIndex[id][0]; ++k) {
531  int flowInID = (int) m_flowInIndex[id][k];
532  Qin += m_Qkin[flowInID]; //m3/s
533  Sin += m_Qsn[flowInID]; // kg/s
534  }
535 
536  // if the channel width is greater than the cell width
537  if (m_streamLink[id] >= 0 && flowwidth <= 0) {
538  m_SedToChannel[id] = Sin * m_TimeStep;
539  return;
540  }
541 
542  //m_Qlastt[id] = m_OverlandQ[id]; // for the next time step

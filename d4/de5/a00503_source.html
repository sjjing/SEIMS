113 int SRD_MB::Execute() {
114  this->CheckInputData();
115 
116  this->initialOutputs();
117 
118  //the first time
119  if (m_isInitial) {
120  int count = 0;
121  for (int i = 0; i < this->m_nCells; i++) {
122  if ((this->m_tMin[i] + this->m_tMax[i]) / 2 < this->m_tsnow) {
123  this->m_SA[i] = this->m_swe0;
124  count++;
125  } //winter
126  else { this->m_SA[i] = 0.0f; } // other seasons
127  }
128 
129  m_swe = this->m_swe0 * count / this->m_nCells;
130  m_isInitial = false;
131  }
132 
133  if (this->m_lastSWE == -99.0f) this->m_lastSWE = this->m_swe;
134  if (m_swe < 0.01) //all cells have not snow, so snow redistribution is 0.
135  {
136  if (this->m_lastSWE >= 0.01) {
137  for (int rw = 0; rw < this->m_nCells; rw++) {
138  m_SR[rw] = 0.0f;
139  }
140  }
141 
142  this->m_lastSWE = this->m_swe;
143  return true;
144  }
145 
146 
147  //if some cells have snow, start to calculate snow redistribution.
148  if (m_w == NULL) m_w = new float[this->m_nCells];
149  if (m_wt == NULL) //get wt. wt is constant for the watershed, so save it in m_wt to save time.
150  {
151  m_wt = new float[this->m_nCells];
152  for (int rw = 0; rw < this->m_nCells; rw++) {
153  m_wt[rw] = 1.0f + 1 / this->m_k_slope * this->m_slope_wind[rw] +
154  1 / this->m_k_curvature * this->m_curva_wind[rw];
155  }
156  }
157 
158  float totalW = 0.0f;
159  for (int rw = 0; rw < this->m_nCells; rw++) {
160  float wl = 1.0f;
161  float shc = this->m_shc[rw] * 1000;
162  if (this->m_swe < shc) {
163  wl = shc / this->m_shc_crop + this->m_swe * (1 - shc / this->m_shc_crop) / shc;
164  }
165 
166  m_w[rw] = m_wt[rw] * wl; //save w in the m_w array, this save time for following loop.
167  totalW += m_w[rw];
168  }
169 
170  float weight = this->m_nCells / totalW; //weight
171  float u = 0.0f; //average wind speed
172  for (int rw = 0; rw < this->m_wsSize; rw++) u += this->m_ws[rw];
173  u /= this->m_wsSize;
174 
175  for (int rw = 0; rw < this->m_nCells; rw++) {
176  float snow = this->m_SA[rw];
177  float tmean = (this->m_tMin[rw] + this->m_tMax[rw]) /
178  2; //if the temperature is higher than t0, the redistribution is 0
179  if (tmean < this->m_tsnow) snow += (1 + this->m_kblow) * this->m_Pnet[rw];
180  if (snow <
181  0.01) // if snow is lower than a very small positive value (0.01), consider it is equal to 0.
182  {
183  this->m_SR[rw] = 0.0f;
184  continue;
185  }
186 
187  if (tmean >= this->m_t0) {
188  this->m_SR[rw] = 0.0f;
189  continue;
190  }
191 
192  float wr = m_w[rw] * weight;
193  float sap = m_swe * wr;
194  float scp = sap - this->m_SA[rw];
195 
196  float ut = float(this->m_ut0 + 0.0033 * pow((this->m_tMin[rw] + this->m_tMax[rw]) / 2 - this->m_t_wind, 2.0f));
197  float ww = 0.0f;
198  if (u - ut >= this->m_u0) ww = 1.0f;
199  if (u - ut >= 0 && u - ut < this->m_u0) ww = (u - ut) / this->m_u0;
200 
201  float sr = scp * ww;
202 
203  if (abs(sr) <= snow) { this->m_SR[rw] = sr; }
204  else { this->m_SR[rw] = snow; }
205  }
206 
207  this->m_lastSWE = this->m_swe;
208  return 0;
209 }

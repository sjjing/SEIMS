76 void MUSLE_I30::initalOutputs() {
77  if (m_sedimentYield == NULL) m_sedimentYield = new float[this->m_cellSize];
78  if (m_usle_ls == NULL) {
79  float constant = pow(22.13f, 0.4f);
80  m_usle_ls = new float[this->m_cellSize];
81  for (int i = 0; i < this->m_cellSize; i++) {
82  float lambda_i1 = this->m_flowacc[i] * this->m_cellWidth;
83  float lambda_i = lambda_i1 + this->m_cellWidth;
84  float L = pow(lambda_i, 1.4f) - pow(lambda_i1, 1.4f);
85  L /= this->m_cellWidth * constant;
86 
87  float S = pow(this->m_slope[i] / 100.0f / 0.0896f, 1.3f);
88 
89  this->m_usle_ls[i] = L * S;//equation 3 in memo, LS factor
90  }
91  }
92  if (m_slopeForPq == NULL) {
93  m_slopeForPq = new float[this->m_cellSize];
94  for (int i = 0; i < this->m_cellSize; i++) {
95  m_slopeForPq[i] = pow(this->m_slope[i] / 100.0f * 1000.0f, 0.16f);
96  }
97  }
98  m_cellAreaKM = pow(this->m_cellWidth / 1000.0f, 2.0f);
99 
100 
101  //initial average half-hour rainfall fraction for each month
102  if (m_alpha_month == NULL) {
103  m_alpha_month = new float[12];
104 
105  //first smooth the extreme half-hour rainfall values, eq.1:3.2.1 p65
106  for (int i = 0; i < 12; i++) {
107  if (i == 0) { m_alpha_month[i] = this->m_p_stat[11][0] + this->m_p_stat[0][0] + this->m_p_stat[1][0]; }
108  else if (i == 11) {
109  m_alpha_month[i] = this->m_p_stat[10][0] + this->m_p_stat[11][0] + this->m_p_stat[0][0];
110  }
111  else { m_alpha_month[i] = this->m_p_stat[i - 1][0] + this->m_p_stat[i][0] + this->m_p_stat[i + 1][0]; }
112 
113  m_alpha_month[i] /= 3.0f;
114  }
115 
116  //get alpha, eq.1:3.2.2 p66
117  for (int i = 0; i < 12; i++) {
118  float alpha = 0.0f;
119  float mu = this->m_p_stat[i][1] / this->m_p_stat[i][2];
120  alpha = log(0.5f / this->m_rain_yrs / this->m_p_stat[i][2]);
121  alpha *= mu;
122  alpha = exp(m_alpha_month[i] / alpha);
123  alpha = this->m_adj_pkr * (1 - alpha);
124 
125  m_alpha_month[i] = alpha;
126  }
127  }
128 
129  //rndseed(6,j) = 1094585182 gcycl.f Line73
130  m_rndseed = 1094585182;
131 }

133 float MUSLE_I30::getPeakRunoffRate(int cell) {
134  float p = this->m_p[cell];
135  float snomlt = this->m_snowMelt[cell];
136  if (p > snomlt) {
137  p -= snomlt; //minus snow melt
138  } else { p = 0.0f; }
139 
140  struct tm datestruture;
141  LocalTime(m_Date, &datestruture);
142  //localtime_s(&datestruture,&m_Date); //get month
143  float max = 1.0f - exp(-125.0f / (p + 5)); //eq.1:3.2.3 p66
144  float a15 = triangularDistribution(0.02083f, m_alpha_month[datestruture.tm_mon], max,
145  &m_rndseed); //eq.1:3.2.4-5 p67
146 
147  float t = this->m_t_concentration[cell];
148  if (t < 0.1f) t = 0.1f; //make sure t >= 0.1
149  float altc = 1.0f - exp(2.0f * t * log(1.0f - a15)); //eq.2:1.3.19 p111
150  float peakr = altc * this->m_surfaceRunoff[cell] * m_cellAreaKM / 3.6f / t;//eq.2:1.3.20 p111
151 
152  return peakr;
153 }

165 float MUSLE_I30::triangularDistribution(float min, float mean, float max, int *seed) {
166  float at1 = min;
167  float at2 = mean;
168  float at3 = max;
169  int *at4i = seed;
170  float u3, rn, y, b1, b2, x1, xx, yy, amn;
171  float atri;
172 
173  u3 = 0.0f;
174  rn = 0.0f;
175  y = 0.0f;
176  b1 = 0.0f;
177  b2 = 0.0f;
178  x1 = 0.0f;
179  xx = 0.0f;
180  yy = 0.0f;
181  amn = 0.0f;
182 
183  u3 = at2 - at1;
184  rn = aunif(at4i);
185  y = 2.0f / (at3 - at1);
186  b2 = at3 - at2;
187  b1 = rn / y;
188  x1 = y * u3 / 2.0f;
189 
190  if (rn <= x1) {
191  xx = 2.0f * b1 * u3;
192  if (xx <= 0.0f) {
193  yy = 0.0f;
194  } else {
195  yy = sqrt(xx);
196  }
197 
198  atri = yy + at1;
199  } else {
200  xx = b2 * b2 - 2.0f * b2 * (b1 - 0.5f * u3);
201  if (xx <= 0.0f) {
202  yy = 0.0f;
203  } else {
204  yy = sqrt(xx);
205  }
206 
207  atri = at3 - yy;
208  }
209 
210  amn = (at3 + at2 + at1) / 3.0f;
211  atri = atri * at2 / amn;
212 
213  if (atri >= 1.0) return 0.99f;
214  if (atri <= 0.0) return 0.001f;
215  return atri;
216 }

218 int MUSLE_I30::Execute() {
219  this->CheckInputData();
220 
221  this->initalOutputs();
222 
223  for (int i = 0; i < this->m_cellSize; i++) {
224  if (this->m_surfaceRunoff[i] < 0.0001f) { this->m_sedimentYield[i] = 0.0f; }
225  else {
226  float q = getPeakRunoffRate(i); //equation 2 in memo, peak flow
227  float Y = 11.8f * pow(this->m_surfaceRunoff[i] * m_cellAreaKM * 1000.0f * q, 0.56f)
228  * this->m_usle_k[i] * this->m_usle_ls[i] * this->m_usle_c[i] *
229  this->m_usle_p[i]; //equation 1 in memo, sediment yield
230 
231  if (this->m_snowAccumulation[i] > 0.0001f) {
232  Y /= exp(3.0f * this->m_snowAccumulation[i] / 25.4f);
233  } //equation 4 in memo, the snow pack effect
234  this->m_sedimentYield[i] = Y;
235  }
236  }
237  return 0;

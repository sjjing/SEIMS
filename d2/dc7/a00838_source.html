43  @staticmethod
44  def rs(doy, n, lat):
45  """solar radiation, n is sunshine duration"""
46  lat = lat * math.pi / 180.
47  a = 0.25
48  b = 0.5
49  d = HydroClimateUtilClass.dec(doy)
50  w = HydroClimateUtilClass.ws(lat, d)
51  nn = 24. * w / math.pi
52  # Extraterrestrial radiation for daily periods
53  ra = (24. * 60. * 0.082 * HydroClimateUtilClass.dr(doy) / math.pi) * \
54  (w * math.sin(lat) * math.sin(d) + math.cos(lat) * math.cos(d) * math.sin(w))
55  return (a + b * n / nn) * ra

74  @staticmethod
75  def get_time_system_from_data_file(in_file):
76  """Get the time system from the data file. The basic format is:
77  #<time_system> [<time_zone>], e.g., #LOCALTIME 8, #UTCTIME
78  """
79  time_sys = 'LOCALTIME'
80  time_zone = time.timezone / -3600
81  f = open(in_file)
82  for line in f:
83  str_line = line
84  for LF in LFs:
85  if LF in line:
86  str_line = line.split(LF)[0]
87  break
88  if str_line[0] != '#':
89  break
90  if str_line.lower().find('utc') >= 0:
91  time_sys = 'UTCTIME'
92  time_zone = 0
93  break
94  if str_line.lower().find('local') >= 0:
95  line_list = StringClass.split_string(str_line, [','])
96  if len(line_list) == 2 and MathClass.isnumerical(line_list[1]):
97  time_zone = -1 * int(line_list[1])
98  break
99  f.close()
100  return time_sys, time_zone

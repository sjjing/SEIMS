16 int SET_LM::Execute() {
17  CheckInputData();
18 
19  if (m_soilET == NULL) {
20  m_soilET = new float[m_nCells];
21  }
22 
23 #pragma omp parallel for
24  for (int i = 0; i < m_nCells; i++) {
25  m_soilET[i] = 0.0f;
26  if (m_soilT[i] <= m_frozenT) { //if the soil temperature is lower than tFrozen, then ES = 0.
27  continue;
28  }
29 
30  float et2d[2];
31  float depth[2];
32  depth[0] = m_upSoilDepth;
33  depth[1] = m_rootDepth[i] - m_upSoilDepth;
34  if (depth[1] < 0) {
35  ostringstream oss;
36  oss << "The root depth at cell(" << i << ") is " << m_rootDepth[i] <<
37  ", and is less than the upper soil depth (" << m_upSoilDepth << endl;
38  throw ModelException("SET_LM", "Execute", oss.str());
39  }
40 
41  float etDeficiency = m_PET[i] - m_EI[i] - m_ED[i];
42  for (int j = 0; j < m_nSoilLayers; j++) {
43  if (etDeficiency <= 0.f) {
44  break;
45  }
46 
47  et2d[j] = 0.f;
48 
49  if (m_sm[i][j] >= m_fc[i][j]) {
50  et2d[j] = etDeficiency;
51  } else if (m_sm[i][j] >= m_wp[i][j]) {
52  et2d[j] = etDeficiency * (m_sm[i][j] - m_wp[i][j]) / (m_fc[i][j] - m_wp[i][j]);
53  } else {
54  et2d[j] = 0.0f;
55  }
56 
57  float sm0 = m_sm[i][j];
58  float availableWater = (m_sm[i][j] - m_wp[i][j]) * depth[j];
59  if (et2d[j] > availableWater) {
60  et2d[j] = availableWater;
61  m_sm[i][j] = m_wp[i][j];
62  } else {
63  m_sm[i][j] -= et2d[j] / depth[j];
64  }
65 
66  if (m_sm[i][j] < 0.f) {
67  cout << "SET_LM\t" << sm0 << "\t" << m_wp[i][j] << "\t" << m_sm[i][j] << "\t"
68  << availableWater / depth[j] << "\t" << et2d[j] / depth[j] << endl;
69  throw ModelException("SET_LM", "Execute", "moisture is less than zero.");
70  }
71 
72  etDeficiency -= et2d[j];
73 
74  m_soilET[i] += et2d[j];
75  }
76  }
77 
78  return 0;
79

214 void ModuleFactory::ReadDLL(string &id, string &dllID) {
215  // the dll file is already read, return
216  if (m_instanceFuncs.find(id) != m_instanceFuncs.end()) {
217  return;
218  }
219  // check if the dll file exists
220  string moduleFileName = trim(m_modulePath) + SEP + string(dllID) + string(Tag_DyLib);
221  if (!FileExists(moduleFileName)) {
222  throw ModelException("ModuleFactory", "ReadDLL", moduleFileName + " does not exist or has no read permission!");
223  }
224 
225  //load library
226 #ifdef windows
227  HINSTANCE handle = LoadLibrary(TEXT(moduleFileName.c_str()));
228  if (handle == NULL) throw ModelException("ModuleFactory", "ReadDLL", "Could not load " + moduleFileName);
229  m_instanceFuncs[id] = InstanceFunction(GetProcAddress(HMODULE(handle), "GetInstance"));
230  m_metadataFuncs[id] = MetadataFunction(GetProcAddress(HMODULE(handle), "MetadataInformation"));
231 #else
232  void *handle = dlopen(moduleFileName.c_str(), RTLD_LAZY);
233  if (handle == NULL) {
234  cout << dlerror() << endl;
235  throw ModelException("ModuleFactory", "ReadDLL", "Could not load " + moduleFileName);
236  }
237  m_instanceFuncs[id] = InstanceFunction(dlsym(handle, "GetInstance"));
238  m_metadataFuncs[id] = MetadataFunction(dlsym(handle, "MetadataInformation"));
239 #endif /* windows */
240  m_dllHandles.push_back(handle);
241  if (m_instanceFuncs[id] == NULL) {
242  throw ModelException("ModuleFactory", "ReadDLL",
243  moduleFileName + " does not implement API function: GetInstance");
244  }
245  if (m_metadataFuncs[id] == NULL) {
246  throw ModelException("ModuleFactory", "ReadDLL",
247  moduleFileName + " does not implement API function: MetadataInformation");
248  }
249  StatusMessage(("Read DLL: " + moduleFileName).c_str());
250 }

596 bool ModuleFactory::LoadSettingsFromFile(const char* filename, vector<vector<string>>& settings) {
597  vector<string> cfgStrs;
598  if (!LoadPlainTextFile(filename, cfgStrs)) {
599  return false;
600  }
601  string T_variables[7] = {DataType_Precipitation, DataType_MeanTemperature, DataType_MaximumTemperature,
602  DataType_MinimumTemperature, DataType_SolarRadiation, DataType_WindSpeed,
603  DataType_RelativeAirMoisture};
604  for (vector<string>::iterator iter = cfgStrs.begin(); iter != cfgStrs.end(); ++iter) {
605  // parse the line into separate item
606  vector<string> tokens = SplitString(*iter, '|');
607  // is there anything in the token list?
608  if (tokens.size() > 0) {
609  for (size_t i = 0; i < tokens.size(); i++) {
610  //TrimSpaces(tokens[i]);
611  tokens[i] = trim(tokens[i]);
612  }
613  // is there anything in the first item?

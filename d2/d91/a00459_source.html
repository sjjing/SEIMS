160 int ExcessRunoff::Execute(void) {
161  CheckInputData();
162 
163  // allocate the output variable
164  if (this->m_pe == NULL) {
165  this->m_pe = new float[this->m_nCells];
166  this->m_infil = new float[this->m_nCells];
167  m_soilMoisture = new float[m_nCells];
168 #pragma omp parallel for
169  for (int i = 0; i < m_nCells; i++) {
170  m_initSoilMoisture[i] = m_initSoilMoisture[i] * m_fieldCap[i];
171  m_soilMoisture[i] = m_initSoilMoisture[i];
172  }
173  }
174 
175 #pragma omp parallel for
176  for (int i = 0; i < m_nCells; i++) {
177  float snm = 0.f;
178  if (m_snowMelt != NULL) {
179  snm = m_snowMelt[i];
180  }
181 
182  float pNet = m_pNet[i];
183  //account for the effects of snow melt and soil temperature
184  float t = m_tMax[i] + m_tMin[i];
185  // snow, without snow melt
186  if (t <= m_tSnow) {
187  pNet = 0.0f;
188  }
189  // rain on snow, no snow melt
190  else if (t > m_tSnow && t <= m_t0 && m_snowAccu[i] > pNet) {
191  pNet = 0.0f;
192  } else {
193  pNet = m_pNet[i] + m_sd[i] + snm;
194  }
195 
196  if (pNet > 0) {
197  // for frozen soil
198  if (m_soilTemp[i] <= m_tFrozen && m_soilMoisture[i] >= m_sFrozen * m_porosity[i]) {
199  m_pe[i] = m_pNet[i] + snm;
200  m_infil[i] = 0.0f;
201  }
202  // for saturation overland flow
203  if (m_soilMoisture[i] > m_porosity[i]) {
204  m_pe[i] = m_pNet[i] + snm;
205  m_infil[i] = 0.0f;
206  } else {
207  float ks = m_ks[i] / 3600.f; // mm/h -> mm/s
208  float limitContent = m_rootDepth[i] * (m_porosity[i] - m_soilMoisture[i]);
209 
210  if (limitContent <= 0.f) {
211  m_infil[i] = 0.f;
212  }
213  //else if(m_soilMoisture[i] >= m_fieldCap[i])
214  //{
215  // m_infil[i] = min(ks*m_dt, limitContent);
216  // m_infil[i] = min(pNet, m_infil[i]);
217  //}
218  else {
219  m_infil[i] = min(pNet, limitContent);
220  }
221 
222  // update excess precipitation
223  m_pe[i] = m_pNet[i] + snm - m_infil[i];
224 
225  // adjust soil moisture
226  m_soilMoisture[i] += m_infil[i] / m_rootDepth[i];
227  }
228  } else {
229  m_pe[i] = 0.0f;
230  m_infil[i] = 0.0f;
231  }
232 
233  }
234 
235  return 0;
236 }

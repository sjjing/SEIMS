291 def get_potential_bmps(suitbmps, sptag, up_sid, up_gvalue, down_sid, down_gvalue, method=1):
292  bmps = suitbmps[sptag][:]
293  bmps = list(set(bmps))
294  if 0 not in bmps:
295  bmps.append(0)
296  if method == 1: # Without any special rule
297  pass
298  elif method == 2:
299  # If not bottom slppos and the downslope unit is configured BMP, then remove 0
300  if down_sid > 0 and down_gvalue == 0:
301  bmps.remove(0)
302  elif method == 3:
303  new_bmps = list()
304  if up_sid < 0 and down_gvalue > 0: # the top slppos, and downslope with BMP
305  for _bid in bmps:
306  if _bid <= down_gvalue:
307  new_bmps.append(_bid)
308  elif down_sid < 0 and up_gvalue > 0: # the bottom slppos, and upslope with BMP
309  for _bid in bmps:
310  if up_gvalue <= _bid:
311  new_bmps.append(_bid)
312  elif down_sid > 0 and up_sid > 0: # middle slppos
313  for _bid in bmps:
314  if down_gvalue == 0 and up_gvalue <= _bid:
315  new_bmps.append(_bid)
316  elif up_gvalue <= _bid <= down_gvalue:
317  new_bmps.append(_bid)
318  else: # Do nothing
319  pass
320  if len(new_bmps) > 0:
321  bmps = list(set(new_bmps))
322  return bmps

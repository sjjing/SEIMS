529 float KinWavSed_CH::GetTransportCapacity(int iReach, int iCell, int id) {
530  // using Beasley et al. (1980) equation for transport capacity of sediment in channel flow
531  float q, S0, K, TranCap, chVol;
532  //WH, CalcuVelocityChannelFlow(iReach, iCell, id);
533  //WH = m_ChannelWH[iReach][iCell];
534  //q = m_ChV[iReach][iCell]* WH * 60;
535  q = m_ChQkin[iReach][iCell] * 60; // convert to m3/min
536  float s = max(0.01f, m_Slope[id]);
537  S0 = sin(atan(s));
538  K = m_USLE_K[id];
539  chVol = m_ChVol[iReach][iCell];

557 void KinWavSed_CH::GetSedimentInFlow(int iReach, int iCell, int id) {
558  float TC, Df, SedtoCh, Deposition, concentration, chVol;
559  //float sedinf;
560  TC = GetTransportCapacity(iReach, iCell, id); //kg/m3
561  m_cap[id] = TC;
562  CalcuChFlowDetachment(iReach, iCell, id);
563  Df = m_CHDETFlow[iReach][iCell]; //kg
564  SedtoCh = m_SedToChannel[id]; //kg
565  m_CHSed_kg[iReach][iCell] += (Df + SedtoCh); //sediment in flow
566  //while the sediment in flow is more than the capacity of the channel flow,
567  // then the sediment deposition would happen; if the velocity of flow is reduced, the TC is decreased correspondingly,
568  // then, the sediment deposition may take place.
569  chVol = m_ChVol[iReach][iCell];
570  if (chVol > 0.0f) {
571  concentration = m_CHSed_kg[iReach][iCell] / chVol; //kg/m3
572  } else {
573  concentration = 0.0f;
574  }
575  Deposition = max(concentration - TC, 0.0f); //kg, >0
576  if (Deposition > 0) {
577  m_CHSed_kg[iReach][iCell] = TC * chVol;
578  }
579  m_CHSedDep[iReach][iCell] = Deposition * chVol; //kg/cell
580  m_depCh[id] = m_CHSedDep[iReach][iCell];
581 }

583 float KinWavSed_CH::simpleSedCalc(float Qn, float Qin, float Sin, float dt, float vol, float sed) {
584  float Qsn = 0;

586  float totwater = vol + Qin * dt; // add upstream water to volume water in cell
587  if (totwater <= 1e-10) {
588  return (Qsn);

590  Qsn = min(totsed / dt, Qn * totsed / totwater);
591  return (Qsn); // sedoutflow is new concentration * new out flux

595 float KinWavSed_CH::complexSedCalc(float Qj1i1, float Qj1i, float Qji1, float Sj1i, float Sji1, float alpha, float dt,
596  float dx) {
597  float Sj1i1, Cavg, Qavg, aQb, abQb_1, A, B, C, s = 0;
598  const float beta = 0.6f;
599 
600  if (Qj1i1 < 1e-6) {
601  return (0);
602  }
603 
604  Qavg = 0.5f * (Qji1 + Qj1i);
605  if (Qavg <= 1e-6) {
606  return (0);
607  }
608 
609  Cavg = (Sj1i + Sji1) / (Qj1i + Qji1);
610  aQb = alpha * pow(Qavg, beta);
611  abQb_1 = alpha * beta * pow(Qavg, beta - 1);
612 
613  A = dt * Sj1i;
614  B = -dx * Cavg * abQb_1 * (Qj1i1 - Qji1);
615  C = (Qji1 <= 1e-6 ? 0 : dx * aQb * Sji1 / Qji1);
616  if (Qj1i1 > 1e-6) {
617  Sj1i1 = (dx * dt * s + A + C + B) / (dt + dx * aQb / Qj1i1); //why s = 0 ?
618  } else {
619  Sj1i1 = 0;
620  }
621 
622  return max(0.0f, Sj1i1);
623 }

31 
32  if (m_startDate <= 0 || m_endDate <= 0) return false;
33 
34  // make sure the start and end times are in the proper order
35  if (m_endDate < m_startDate) {
36  time_t tmp = m_startDate;
37  m_startDate = m_endDate;
38  m_endDate = tmp;
39  }
40 
41  m_mode = GetUpper(GetValue(Tag_Mode));
42 
43  //read interval
44  vector<string> dtList = SplitString(GetValue(Tag_Interval), ',');
45  m_dtHs = atoi(dtList[0].c_str());
46  m_dtCh = m_dtHs;

50  // convert the time interval to seconds to conform to time_t struct
51  if (StringMatch(m_mode, Tag_Mode_Daily)) {

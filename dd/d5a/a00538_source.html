21 int FindBoundingBox(clsRasterData<int> &rsSubbasin, map<int, SubBasin> &bboxMap) {

23  int nYSize = rsSubbasin.getRows();
24 
25  int *pData = rsSubbasin.getRasterDataPointer();
26  int noDataValue = rsSubbasin.getNoDataValue();
27  // find bounding box for each subbasin
28  for (int i = 0; i < nYSize; i++) {
29  for (int j = 0; j < nXSize; j++) {
30  int id = pData[i * nXSize + j];
31  if (noDataValue != id) {
32  if (bboxMap.find(id) == bboxMap.end()) {
33  bboxMap[id] = SubBasin(j, i, j, i);
34  } else {
35  if (j < bboxMap[id].xMin) {
36  bboxMap[id].xMin = j;
37  } else if (j > bboxMap[id].xMax) {
38  bboxMap[id].xMax = j;
39  }
40 
41  if (i > bboxMap[id].yMax) {
42  bboxMap[id].yMax = i;
43  }
44  }
45  bboxMap[id].cellCount += 1;
46  }
47  }
48  }
49  return 0;
50 }

47  //if(temp <= 0.0f)
48  // temp = 0.001f;
49 
50  m_recharge[i] = 0.f;
51  if (moisture > m_FieldCapacity[i]) {
52  // the water exceeds the porosity is added to percolation directly
53  if (moisture > m_Porosity[i]) {
54  m_recharge[i] += (moisture - m_Porosity[i]) * m_rootDepth[i];
55  m_Moisture[i] = m_Porosity[i];
56  }
57 
58  // recharge capacity (mm)
59  float dcIndex = 3.f + 2.f / m_Poreindex[i]; // pore disconnectedness index
60  //float rechargeCap = m_Conductivity[i] / 3600.f * m_timestep * pow((moisture - m_Residual[i])/temp, dcIndex);
61  float rechargeCap =
62  m_Conductivity[i] / 3600.f * m_timestep * pow(moisture / m_Porosity[i], dcIndex); //Campbell, 1974
63  float availableWater = (m_Moisture[i] - m_FieldCapacity[i]) * m_rootDepth[i];
64  if (rechargeCap >= availableWater) {
65  rechargeCap = availableWater;
66  }
67 
68  m_recharge[i] += rechargeCap;
69  m_Moisture[i] -= m_recharge[i] / m_rootDepth[i];
70  }
71  }

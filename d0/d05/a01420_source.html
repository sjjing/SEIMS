364 void NutrientTransportSediment::OrgNRemovedInRunoff_StaticMethod(int i) {
365  //amount of organic N in first soil layer (orgninfl)
366  float orgninfl = 0.f;
367  //conversion factor (wt)
368  float wt = 0.f;
369  orgninfl = m_sol_orgn[i][0] + m_sol_aorgn[i][0] + m_sol_fon[i][0];
370  wt = m_sol_bd[i][0] * m_soilThick[i][0] / 100.f;
371  //concentration of organic N in soil (concn)
372  float concn = 0.f;
373  concn = orgninfl * m_enratio[i] / wt;
374  //Calculate the amount of organic nitrogen transported with sediment to the stream, equation 4:2.2.1 in SWAT Theory 2009, p271
375  m_sedorgn[i] = 0.001f * concn * m_sedEroded[i] / 1000.f / m_cellArea;
376  //update soil nitrogen pools
377  if (orgninfl > 1.e-6f) {
378  m_sol_aorgn[i][0] = m_sol_aorgn[i][0] - m_sedorgn[i] * (m_sol_aorgn[i][0] / orgninfl);
379  m_sol_orgn[i][0] = m_sol_orgn[i][0] - m_sedorgn[i] * (m_sol_orgn[i][0] / orgninfl);
380  m_sol_fon[i][0] = m_sol_fon[i][0] - m_sedorgn[i] * (m_sol_fon[i][0] / orgninfl);
381  if (m_sol_aorgn[i][0] < 0.f) {
382  m_sedorgn[i] = m_sedorgn[i] + m_sol_aorgn[i][0];
383  m_sol_aorgn[i][0] = 0.f;
384  }
385  if (m_sol_orgn[i][0] < 0.f) {
386  m_sedorgn[i] = m_sedorgn[i] + m_sol_orgn[i][0];
387  m_sol_orgn[i][0] = 0.f;
388  }
389  if (m_sol_fon[i][0] < 0.f) {
390  m_sedorgn[i] = m_sedorgn[i] + m_sol_fon[i][0];
391  m_sol_fon[i][0] = 0.f;
392  }
393  }

400 void NutrientTransportSediment::OrgNRemovedInRunoff_CENTURY(int i) {
401  float totOrgN_lyr0 = 0.f;
402  float wt1 = 0.f;
403  float er = 0.f;
404  float conc = 0.f;
405  float sol_mass = 0.f;

409  float YOC = 0.f;

412  float YEW = 0.f;
413  float X1 = 0.f, PRMT_21 = 0.f;
414  float PRMT_44 = 0.f;
415  float XX = 0.f, DK = 0.f, V = 0.f, X3 = 0.f;

422  er = m_enratio[i];
423  conc = totOrgN_lyr0 * er / wt1;
424  m_sedorgn[i] = 0.001f * conc * m_sedEroded[i] / 1000.f / m_cellArea;
426  if (totOrgN_lyr0 > UTIL_ZERO) {
427  float xx1 = (1.f - m_sedorgn[i] / totOrgN_lyr0);
428  m_sol_LSN[i][0] *= xx1;
429  m_sol_LMN[i][0] *= xx1;
430  m_sol_HPN[i][0] *= xx1;
431  m_sol_HSN[i][0] *= xx1;
432  }
434  sol_mass = m_soilThick[i][0] / 1000.f * 10000.f * m_sol_bd[i][0] * 1000.f *
435  (1.f - m_sol_rock[i][0] / 100.f);
436  TOT = m_sol_HPC[i][0] + m_sol_HSC[i][0] + m_sol_LMC[i][0] + m_sol_LSC[i][0];
439  YEW = min((m_sedEroded[i] / m_cellArea + YW / m_cellArea) / sol_mass, 0.9f);
440 
441  X1 = 1.f - YEW;
442  YOC = YEW * TOT;
443  m_sol_HSC[i][0] *= X1;
444  m_sol_HPC[i][0] *= X1;
445  m_sol_LS[i][0] *= X1;
446  m_sol_LM[i][0] *= X1;
447  m_sol_LSL[i][0] *= X1;
448  m_sol_LSC[i][0] *= X1;
449  m_sol_LMC[i][0] *= X1;
450  m_sol_LSLC[i][0] *= X1;
451  m_sol_LSLNC[i][0] = m_sol_LSC[i][0] - m_sol_LSLC[i][0];
452 
453  if (m_sol_BMC[i][0] > 0.01f) {
455  PRMT_21 = 1000.f;
456  m_sol_WOC[i][0] = m_sol_LSC[i][0] + m_sol_LMC[i][0] + m_sol_HPC[i][0] + m_sol_HSC[i][0] + m_sol_BMC[i][0];
457  DK = 0.0001f * PRMT_21 * m_sol_WOC[i][0];
458  X1 = m_sol_wsatur[i][0];
459  if (X1 <= 0.f) X1 = 0.01f;
460  XX = X1 + DK;
461  V = m_surfaceRunoff[i] + m_sol_perco[i][0] + m_sol_laterq[i][0];
462  if (V > 1.e-10f) {
463  X3 = m_sol_BMC[i][0] * (1.f - exp(-V / XX));
464  PRMT_44 = 0.5;
465  CO = X3 / (m_sol_perco[i][0] + PRMT_44 * (m_surfaceRunoff[i] + m_sol_laterq[i][0]));
466  CS = PRMT_44 * CO;
467  VBC = CO * m_sol_perco[i][0];
468  m_sol_BMC[i][0] -= X3;
469  QBC = CS * (m_surfaceRunoff[i] + m_sol_laterq[i][0]);
471  if (YEW > 0.f) {
472  CS = DK * m_sol_BMC[i][0] / XX;
473  YBC = YEW * CS;
474  }
475  }
476  }
477  m_sol_BMC[i][0] -= YBC;
479  m_sol_latC[i][0] = QBC * (m_sol_laterq[i][0] / (m_surfaceRunoff[i] + m_sol_laterq[i][0] + UTIL_ZERO));
480  m_sol_percoC[i][0] = VBC;
481  m_sedCLoss[i] = YOC + YBC;
482 
483  latc_clyr += m_sol_latC[i][0];
484  for (int k = 1; k < (int) m_nSoilLayers[i]; k++) {
485  m_sol_WOC[i][k] = m_sol_LSC[i][k] + m_sol_LMC[i][k] + m_sol_HPC[i][k] + m_sol_HSC[i][k];
486  float Y1 = m_sol_BMC[i][k] + VBC;
487  VBC = 0.f;
488  if (Y1 > 0.01f) {
489  V = m_sol_perco[i][k] + m_sol_laterq[i][k];
490  if (V > 0.f) {
491  VBC = Y1 * (1.f - exp(-V / (m_sol_wsatur[i][k] + 0.0001f * PRMT_21 * m_sol_WOC[i][k])));
492  }
493  }
494  m_sol_latC[i][k] = VBC * (m_sol_laterq[i][k] / (m_sol_laterq[i][k] + m_sol_perco[i][k] + UTIL_ZERO));
495  m_sol_percoC[i][k] = VBC - m_sol_latC[i][k];
496  m_sol_BMC[i][k] = Y1 - VBC;
497 
499  perc_clyr += m_sol_percoC[i][k];
500  latc_clyr += m_sol_latC[i][k];
501  }
502  m_laterC[i] = latc_clyr;
503  m_percoC[i] = perc_clyr;

506 void NutrientTransportSediment::OrgPAttachedtoSed(int i) {
507  //amount of phosphorus attached to sediment in soil (sol_attp)
508  float sol_attp = 0.f;
509  //fraction of active mineral/organic/stable mineral phosphorus in soil (sol_attp_o, sol_attp_a, sol_attp_s)
510  float sol_attp_o = 0.f;
511  float sol_attp_a = 0.f;
512  float sol_attp_s = 0.f;
513  //Calculate sediment
514  sol_attp = m_sol_orgp[i][0] + m_sol_fop[i][0] + m_sol_actp[i][0] + m_sol_stap[i][0];
515  if (m_sol_mp != NULL) {
516  sol_attp += m_sol_mp[i][0];
517  }
518  if (sol_attp > 1.e-3f) {
519  sol_attp_o = (m_sol_orgp[i][0] + m_sol_fop[i][0]) / sol_attp;
520  if (m_sol_mp != NULL) {
521  sol_attp_o += m_sol_mp[i][0] / sol_attp;
522  }
523  sol_attp_a = m_sol_actp[i][0] / sol_attp;
524  sol_attp_s = m_sol_stap[i][0] / sol_attp;
525  }
526  //conversion factor (mg/kg => kg/ha) (wt)
527  float wt = m_sol_bd[i][0] * m_soilThick[i][0] / 100.f;
528  //concentration of organic P in soil (concp)
529  float concp = 0.f;
530  concp = sol_attp * m_enratio[i] / wt;
531  //total amount of P removed in sediment erosion (sedp)
532  float sedp = 1.e-6f * concp * m_sedEroded[i] / m_cellArea;
533  m_sedorgp[i] = sedp * sol_attp_o;
534  m_sedminpa[i] = sedp * sol_attp_a;
535  m_sedminps[i] = sedp * sol_attp_s;
536 
537  //if(i==100)cout << "sedp: " << sedp<< ",sol_attp_o: " << sol_attp_o << endl;
538  //modify phosphorus pools
539 
540  //total amount of P in mineral sediment pools prior to sediment removal (psedd) // Not used
541  //float psedd = 0.f;
542  //psedd = m_sol_actp[i][0] + m_sol_stap[i][0];
543 
544  //total amount of P in organic pools prior to sediment removal (porgg)
545  float porgg = 0.f;
546  porgg = m_sol_orgp[i][0] + m_sol_fop[i][0];
547  if (porgg > 1.e-3f) {
548  m_sol_orgp[i][0] = m_sol_orgp[i][0] - m_sedorgp[i] * (m_sol_orgp[i][0] / porgg);
549  m_sol_fop[i][0] = m_sol_fop[i][0] - m_sedorgp[i] * (m_sol_fop[i][0] / porgg);
550  }
551  m_sol_actp[i][0] = m_sol_actp[i][0] - m_sedminpa[i];
552  m_sol_stap[i][0] = m_sol_stap[i][0] - m_sedminps[i];
553  if (m_sol_orgp[i][0] < 0.f) {
554  m_sedorgp[i] = m_sedorgp[i] + m_sol_orgp[i][0];
555  m_sol_orgp[i][0] = 0.f;
556  }
557  if (m_sol_fop[i][0] < 0.f) {
558  m_sedorgp[i] = m_sedorgp[i] + m_sol_fop[i][0];
559  m_sol_fop[i][0] = 0.f;
560  }
561  if (m_sol_actp[i][0] < 0.f) {
562  m_sedminpa[i] = m_sedminpa[i] + m_sol_actp[i][0];
563  m_sol_actp[i][0] = 0.f;
564  }
565  if (m_sol_stap[i][0] < 0.f) {
566  m_sedminps[i] = m_sedminps[i] + m_sol_stap[i][0];
567  m_sol_stap[i][0] = 0.f;
568  }
569  // if (i == 46364) cout << "surfq: " <<m_surfaceRunoff[i]<< ", sedYld: "<<m_sedEroded[i]<<", sedorgp: "<<m_sedorgp[i]<< endl;

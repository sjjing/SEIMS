320  m_fileIn1DStrs.resize(sz + 1); // resize with one more row
321  m_fileIn1DStrs[sz] = tokens[0] + "|" + tokens[1]; // keep the interface consistent
322  }
323  bson_destroy(b);
324  mongoc_cursor_destroy(cursor);

437  }

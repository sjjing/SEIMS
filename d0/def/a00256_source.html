53 int *GetOutDegreeMatrix(const int *dirMatrix, int nRows, int nCols, int dirNoDataValue) {
54  int n = nRows * nCols;
55  int *outDegreeMatrix = new int[n];
56  int index = 0;
57  for (int i = 0; i < nRows; i++) {
58  for (int j = 0; j < nCols; j++) {
59  index = i * nCols + j;
60  outDegreeMatrix[index] = 0;
61 
62  int flow_dir = dirMatrix[index];
63  if (flow_dir == dirNoDataValue || flow_dir < 0) {
64  outDegreeMatrix[index] = dirNoDataValue;
65  continue;
66  }
67 
68  if ((flow_dir & 1) && (j != nCols - 1) && (dirMatrix[index + 1] != dirNoDataValue) &&
69  dirMatrix[index + 1] > 0) {
70  outDegreeMatrix[index]++;
71  }
72 
73  if ((flow_dir & 2) && (i != nRows - 1) && (j != nCols - 1) &&
74  (dirMatrix[index + nCols + 1] != dirNoDataValue) && dirMatrix[index + nCols + 1] > 0) {
75  outDegreeMatrix[index]++;
76  }
77 
78  if ((flow_dir & 4) && (i != nRows - 1) && (dirMatrix[index + nCols] != dirNoDataValue) &&
79  dirMatrix[index + nCols] > 0) {
80  outDegreeMatrix[index]++;
81  }
82 
83  if ((flow_dir & 8) && (i != nRows - 1) && (j != 0) && (dirMatrix[index + nCols - 1] != dirNoDataValue) &&
84  dirMatrix[index + nCols - 1] > 0) {
85  outDegreeMatrix[index]++;
86  }
87 
88  if ((flow_dir & 16) && (j != 0) && (dirMatrix[index - 1] != dirNoDataValue) && dirMatrix[index - 1] > 0) {
89  outDegreeMatrix[index]++;
90  }
91 
92  if ((flow_dir & 32) && (i != 0) && (j != 0) && (dirMatrix[index - nCols - 1] != dirNoDataValue) &&
93  dirMatrix[index - nCols - 1] > 0) {
94  outDegreeMatrix[index]++;
95  }
96 
97  if ((flow_dir & 64) && (i != 0) && (dirMatrix[index - nCols] != dirNoDataValue) &&
98  dirMatrix[index - nCols] > 0) {
99  outDegreeMatrix[index]++;
100  }
101 
102  if ((flow_dir & 128) && (i != 0) && (j != nCols - 1) && (dirMatrix[index - nCols + 1] != dirNoDataValue) &&
103  dirMatrix[index - nCols + 1] > 0) {
104  outDegreeMatrix[index]++;
105  }
106  }
107  }
108  return outDegreeMatrix;
109 }

111 void UpdateUpStream(int i, int j, int nRows, int nCols, const int *dirMatrix, int *outDegreeMatrix,
112  int &numNextLayer, int *nextLayer) {
113  int index = i * nCols + j + 1;
114  if (j != nCols - 1 && dirMatrix[index] & 16) {
115  if ((--outDegreeMatrix[index]) == 0) {
116  nextLayer[numNextLayer++] = index;
117  }
118  }
119 
120  index = (i + 1) * nCols + j + 1;
121  if (i != nRows - 1 && j != nCols - 1 && dirMatrix[index] & 32) {
122  if ((--outDegreeMatrix[index]) == 0) {
123  nextLayer[numNextLayer++] = index;
124  }
125  }
126 
127  index = (i + 1) * nCols + j;
128  if (i != nRows - 1 && dirMatrix[index] & 64) {
129  if ((--outDegreeMatrix[index]) == 0) {
130  nextLayer[numNextLayer++] = index;
131  }
132  }
133 
134  index = (i + 1) * nCols + j - 1;
135  if (i != nRows - 1 && j != 0 && dirMatrix[index] & 128) {
136  if ((--outDegreeMatrix[index]) == 0) {
137  nextLayer[numNextLayer++] = index;
138  }
139  }
140 
141  index = i * nCols + j - 1;
142  if (j != 0 && dirMatrix[index] & 1) {
143  if ((--outDegreeMatrix[index]) == 0) {
144  nextLayer[numNextLayer++] = index;
145  }
146  }
147 
148  index = (i - 1) * nCols + j - 1;
149  if (i != 0 && j != 0 && dirMatrix[index] & 2) {
150  if ((--outDegreeMatrix[index]) == 0) {
151  nextLayer[numNextLayer++] = index;
152  }
153  }
154 
155  index = (i - 1) * nCols + j;
156  if (i != 0 && dirMatrix[index] & 4) {
157  if ((--outDegreeMatrix[index]) == 0) {
158  nextLayer[numNextLayer++] = index;
159  }
160  }
161 
162  index = (i - 1) * nCols + j + 1;
163  if (i != 0 && j != nCols - 1 && dirMatrix[index] & 8) {
164  if ((--outDegreeMatrix[index]) == 0) {
165  nextLayer[numNextLayer++] = index;
166  }
167  }

170 string GridLayeringFromOutlet(int nValidGrids, const int *dirMatrix, const int *compressedIndex, int *outDegreeMatrix,
171  int nRows, int nCols, int dirNoDataValue, int outputNoDataValue, int *&layeringNum) {
172  int n = nRows * nCols;
173  layeringNum = new int[n];
174 
175  int *lastLayer = new int[n];
176  int numLastLayer = 0;
177 
178  bool flag = true;
179  int curNum = 0;
180  int index = 0;
181 
182 
183  // 1. find outlet grids
184  for (int i = 0; i < nRows; i++) {
185  for (int j = 0; j < nCols; j++) {
186  index = i * nCols + j;
187 
188  if (dirMatrix[index] == dirNoDataValue) {
189  layeringNum[index] = outputNoDataValue;
190  } else {
191  if (outDegreeMatrix[index] == 0) {
192  lastLayer[numLastLayer++] = index;
193  }
194  }
195  }
196  }
197 
198  string strLayers = "";
199  //cout << "nubmer of outlets: " << numLastLayer << endl;
200 
201  // 2. loop and layer
202  int numNextLayer = 0;
203  int *nextLayer = new int[n];
204  int *tmp;
205 
206  while (numLastLayer > 0) {
207  curNum++;
208 
209  ostringstream oss;
210  oss << "\n" << numLastLayer << "\t";
211 
212  numNextLayer = 0;
213  for (int iInLayer = 0; iInLayer < numLastLayer; iInLayer++) {
214  index = lastLayer[iInLayer];
215  oss << compressedIndex[index] << "\t";
216 
217  layeringNum[index] = curNum;
218 
219  int i = index / nCols;
220  int j = index % nCols;
221 
222  UpdateUpStream(i, j, nRows, nCols, dirMatrix, outDegreeMatrix, numNextLayer, nextLayer);
223  }
224  strLayers = oss.str() + strLayers;
225 
226  numLastLayer = numNextLayer;
227 
228  tmp = lastLayer;
229  lastLayer = nextLayer;
230  nextLayer = tmp;
231  }
232 
233  ostringstream outputOss;
234  outputOss << nValidGrids << "\t" << curNum << strLayers;
235  // 3. reverse the layer number
236  for (int i = 0; i < n; i++) {
237  if (layeringNum[i] != outputNoDataValue) {
238  layeringNum[i] = curNum - layeringNum[i] + 1;
239  }
240  }
241 
242  delete[] lastLayer;
243  delete[] nextLayer;
244  lastLayer = NULL;
245  nextLayer = NULL;
246 
247  return outputOss.str();
248 }

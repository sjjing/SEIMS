19 def OGRWkt2Shapely(input_shape, idField):
20  shapely_objects = []
21  idList = []
22  # print input_shape
23  shp = ogr.Open(input_shape)
24  lyr = shp.GetLayer()
25  for n in range(0, lyr.GetFeatureCount()):
26  feat = lyr.GetFeature(n)
27  wkt_feat = loads(feat.geometry().ExportToWkt())
28  shapely_objects.append(wkt_feat)
29  idIndex = feat.GetFieldIndex(idField)
30  idList.append(feat.GetField(idIndex))
31  return shapely_objects, idList

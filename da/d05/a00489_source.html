87 int clsPI_MSM::Execute() {
88  //check input data
89  CheckInputData();
91  initialOutputs();
92 
93  int julian = JulianDay(m_date);
94 #pragma omp parallel for
95  for (int i = 0; i < this->m_nCells; i++) {
96  if (m_P[i] > 0.f) {
97  //interception storage capacity
98  float degree = 2.f * PI * (julian - 87.f) / 365.f;
100  float min = m_minSt[i];
101  float max = m_maxSt[i];
102  float capacity = min + (max - min) * pow(0.5f + 0.5f * sin(degree), m_Pi_b);
103 
104  //interception, currently, m_st[i] is storage of (t-1) time step
105  float availableSpace = capacity - m_st[i];
106  if (availableSpace < 0) {
107  availableSpace = 0.f;
108  }
109 
110  if (availableSpace < m_P[i]) {
111  m_interceptionLoss[i] = availableSpace;
112  } else {
113  m_interceptionLoss[i] = m_P[i];
114  }
115 
116  //net precipitation
117  m_netPrecipitation[i] = m_P[i] - m_interceptionLoss[i];
118  m_st[i] += m_interceptionLoss[i];
119  } else {
120  m_interceptionLoss[i] = 0.f;
121  m_netPrecipitation[i] = 0.f;
122  }
123  //evaporation
124  if (m_st[i] > m_PET[i]) {
125  m_evaporation[i] = m_PET[i];
126  } else {
127  m_evaporation[i] = m_st[i];
128  }
129  m_st[i] -= m_evaporation[i];
130  }
131  return 0;
132 }

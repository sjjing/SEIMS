105 void InterFlow_IKW::FlowInSoil(int id) {
106  //sum the upstream overland flow
107  float qUp = 0.0f;
108  for (int k = 1; k <= (int) m_flowInIndex[id][0]; ++k) {
109  int flowInID = (int) m_flowInIndex[id][k];
110  if (m_streamLink[id] > 0) {
111  qUp += m_q[flowInID];
112  }
113  }
114 
115  float flowWidth = m_CellWidth;
116  // there is no land in this cell
117  if (m_streamLink[id] > 0) {
118  flowWidth -= m_chWidth[id];
119  if (flowWidth <= 0) {
120  m_q[id] = qUp;
121  m_h[id] = 0.f;
122  }
123  return;
124  }
125 
126  // adjust soil moisture
127  float s0 = m_s0[id];
128  float soilVolumn = m_rootDepth[id] / 1000 * m_CellWidth * flowWidth / cos(atan(s0)); //m3
129  m_soilMoistrue[id] += qUp * m_dt / soilVolumn;
130 
131  // the water exceeds the porosity is added to storage (return flow)
132  if (m_soilMoistrue[id] > m_porosity[id]) {
133  m_hReturnFlow[id] = (m_soilMoistrue[id] - m_porosity[id]) * m_rootDepth[id];
134  m_sr[id] += m_hReturnFlow[id];
135  m_soilMoistrue[id] = m_porosity[id];
136  }
137 
138  // if soil moisture is below the field capacity, no interflow will be generated
139  if (m_soilMoistrue[id] < m_fieldCapacity[id]) {
140  m_q[id] = 0.f;
141  m_h[id] = 0.f;
142  return;
143  }
144 
145  // calculate effective hydraulic conductivity (mm/h -> m/s)
146  //float k = m_ks[id]/1000/3600 * pow((m_soilMoistrue[id] - m_residual[id])/(m_porosity[id] - m_residual[id]), m_poreIndex[id]);
147  float k = m_ks[id] / 1000 / 3600 * pow(m_soilMoistrue[id] / m_porosity[id], m_poreIndex[id]);
148  // calculate interflow (m3/s)
149  m_q[id] = m_landuseFactor * m_rootDepth[id] / 1000 * s0 * k * m_CellWidth;
150 
151  // available water
152  float availableWater = (m_soilMoistrue[id] - m_fieldCapacity[id]) * soilVolumn;
153  float interFlow = m_q[id] * m_dt; // m3
154  if (interFlow > availableWater) {
155  m_q[id] = availableWater / m_dt;
156  interFlow = availableWater;
157  }
158  m_h[id] = 1000 * interFlow / (m_CellWidth * m_CellWidth);
159 
160  // adjust soil moisture
161  m_soilMoistrue[id] -= interFlow / soilVolumn;
162 }

164 int InterFlow_IKW::Execute() {
165 
166  initialOutputs();
167 
168  for (int iLayer = 0; iLayer < m_nLayers; ++iLayer) {
169  // There are not any flow relationship within each routing layer.
170  // So parallelization can be done here.
171  int nCells = (int) m_routingLayers[iLayer][0];
172  //SetOpenMPThread(2);
173 #pragma omp parallel for
174  for (int iCell = 1; iCell <= nCells; ++iCell) {
175  int id = (int) m_routingLayers[iLayer][iCell];
176  FlowInSoil(id);
177  }
178  }
179  return 0;
180 }

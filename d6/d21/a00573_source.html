18 string LayeringFromSourceD8(const char *outputDir, mongoc_gridfs_t *gfs, int id, int nValidGrids, const int *dirMatrix,
19  const int *compressedIndex, RasterHeader &header, int outputNoDataValue) {
20 
21  int nRows = header.nRows;
22  int nCols = header.nCols;
23  int dirNoDataValue = header.noDataValue;
24 
25  // preprocessing
26  /* accumulate each cell's upstream direction
27  * e.g. cell (i, j) has three upstream source: (i, j+1), (i-1, j), (i+1, j)
28  * then the reverseDir value of cell (i, j) is 1 + 64 + 4 = 69
29  */
30  int *reverseDirMatrix = GetReverseDirMatrix(dirMatrix, nRows, nCols, dirNoDataValue);
31  /* count each cell's upstream number by bitwise AND operator
32  * e.g. cell (i, j) has a reversed direction value of 69, which stored as 1000101
33  * 1000101 & 1 is True, and so as to 100, 1000000. So the upstream cell number is 3.
34  */
35  int *inDegreeMatrix = GetInDegreeMatrix(reverseDirMatrix, nRows, nCols, dirNoDataValue);
36 
37  float *flowInOutput;
38  int nFlowInOutput = OutputMultiFlowOut(nRows, nCols, nValidGrids, inDegreeMatrix, reverseDirMatrix, dirNoDataValue,
39  compressedIndex, flowInOutput);
40  // output to txt
41  ostringstream oss;
42  oss << outputDir << "/" << id << "_FLOWIN_INDEX_D8.txt";
43  ofstream ofs_flowInD8(oss.str().c_str());
44  ofs_flowInD8 << flowInOutput[0] << endl;
45  ofs_flowInD8 << "ID\tUpstreamCount\tUpstreamID" << endl;
46  int tmpCount = 1;
47  for (int i = 0; i < flowInOutput[0]; i++) {
48  int upstreamcount = flowInOutput[tmpCount++];
49  ofs_flowInD8 << i << "\t" << upstreamcount << "\t";
50  for (int j = 0; j < upstreamcount; j++) {
51  ofs_flowInD8 << flowInOutput[tmpCount++] << ", ";
52  }
53  ofs_flowInD8 << endl;
54  }
55  ofs_flowInD8.close();
56 
57  WriteStringToMongoDB(gfs, id, "FLOWIN_INDEX_D8", nFlowInOutput, (char *) flowInOutput);
58 
59  // perform grid layering
60  int *layeringGrid = NULL;
61  string outputStr = GridLayeringFromSource(nValidGrids, dirMatrix, compressedIndex, inDegreeMatrix, nRows, nCols,
62  dirNoDataValue, outputNoDataValue, layeringGrid);
63 
64  // output layering result
65  // ostringstream oss;
66  oss << outputDir << "/" << id << "_ROUTING_LAYERS_UP_DOWN.asc";
67  //cout << oss.str().c_str() << endl;
68  OutputArcAscii(oss.str().c_str(), header, layeringGrid, outputNoDataValue);
69 
70  oss.str("");
71  oss << outputDir << "/" << id << "_ROUTING_LAYERS_UP_DOWN.txt";
72  //cout << oss.str().c_str() << endl;
73  ofstream ofs(oss.str().c_str());
74  ofs << outputStr;
75  ofs.close();
76 
77  delete[] reverseDirMatrix;
78  delete[] inDegreeMatrix;
79  delete[] layeringGrid;
80  delete[] flowInOutput;
81  reverseDirMatrix = NULL;
82  inDegreeMatrix = NULL;
83  layeringGrid = NULL;
84  flowInOutput = NULL;
85 
86  cout << "OutputD8FlowInIndexes done!" << endl;
87  return oss.str();
88 }

149 // Get the reverse flow direction matrix
150 // Modified from Zhan Li-Jun 2012
151 int *GetReverseDirMatrix(const int *dirMatrix, int nRows, int nCols, int dirNoDataValue) {
152  int n = nRows * nCols;
153  int *reverseDirMatrix = new int[n];
154 #pragma omp parallel for
155  for (int i = 0; i < n; i++) {
156  reverseDirMatrix[i] = 0;
157  }
158 #pragma omp parallel for
159  for (int i = 0; i < nRows; i++) {
160  for (int j = 0; j < nCols; j++) {
161  int index = i * nCols + j;
162  int flow_dir = dirMatrix[index];
163  if (flow_dir == dirNoDataValue || flow_dir < 0) {
164  reverseDirMatrix[index] = dirNoDataValue;
165  continue;
166  }
167 
168  if ((flow_dir & 1) && j != nCols - 1 && dirMatrix[index + 1] != dirNoDataValue
169  && dirMatrix[index + 1] > 0) {
170  reverseDirMatrix[index + 1] += 16;
171  }
172 
173  if ((flow_dir & 2) && i != nRows - 1 && j != nCols - 1 && dirMatrix[index + nCols + 1] != dirNoDataValue &&
174  dirMatrix[index + nCols + 1] > 0) {
175  reverseDirMatrix[index + nCols + 1] += 32;
176  }
177 
178  if ((flow_dir & 4) && i != nRows - 1 && dirMatrix[index + nCols] != dirNoDataValue &&
179  dirMatrix[index + nCols] > 0) {
180  reverseDirMatrix[index + nCols] += 64;
181  }
182 
183  if ((flow_dir & 8) && i != nRows - 1 && j != 0 && dirMatrix[index + nCols - 1] != dirNoDataValue &&
184  dirMatrix[index + nCols - 1] > 0) {
185  reverseDirMatrix[index + nCols - 1] += 128;
186  }
187 
188  if ((flow_dir & 16) && j != 0 && dirMatrix[index - 1] != dirNoDataValue && dirMatrix[index - 1] > 0) {
189  reverseDirMatrix[index - 1] += 1;
190  }
191 
192  if ((flow_dir & 32) && i != 0 && j != 0 && dirMatrix[index - nCols - 1] != dirNoDataValue &&
193  dirMatrix[index - nCols - 1] > 0) {
194  reverseDirMatrix[index - nCols - 1] += 2;
195  }
196 
197  if ((flow_dir & 64) && i != 0 && dirMatrix[index - nCols] != dirNoDataValue
198  && dirMatrix[index - nCols] > 0) {
199  reverseDirMatrix[index - nCols] += 4;
200  }
201 
202  if ((flow_dir & 128) && i != 0 && j != nCols - 1 && dirMatrix[index - nCols + 1] != dirNoDataValue &&
203  dirMatrix[index - nCols + 1] > 0) {
204  reverseDirMatrix[index - nCols + 1] += 8;
205  }
206  }
207  }
208 
209  return reverseDirMatrix;
210 }

212 // Get the in-degree matrix
213 int *GetInDegreeMatrix(const int *reverseDirMatrix, int nRows, int nCols, int noDataValue) {
214  int n = nRows * nCols;
215  int *inDegreeMatrix = new int[n];
216 #pragma omp parallel for
217  for (int i = 0; i < n; i++) {
218  inDegreeMatrix[i] = 0;
219  int flow_dir = reverseDirMatrix[i];
220  if (flow_dir == noDataValue || flow_dir < 0) {
221  inDegreeMatrix[i] = noDataValue;
222  continue;
223  }
224 
225  if (flow_dir & 1) {
226  inDegreeMatrix[i]++;
227  }
228  if (flow_dir & 2) {
229  inDegreeMatrix[i]++;
230  }
231  if (flow_dir & 4) {
232  inDegreeMatrix[i]++;
233  }
234  if (flow_dir & 8) {
235  inDegreeMatrix[i]++;
236  }
237  if (flow_dir & 16) {
238  inDegreeMatrix[i]++;
239  }
240  if (flow_dir & 32) {
241  inDegreeMatrix[i]++;
242  }
243  if (flow_dir & 64) {
244  inDegreeMatrix[i]++;
245  }
246  if (flow_dir & 128) {
247  inDegreeMatrix[i]++;
248  }
249  }
250 
251  return inDegreeMatrix;
252 }

254 void UpdateDownStream(int i, int j, int nRows, int nCols, const int *dirMatrix, int *inDegreeMatrix,
255  int &numNextLayer, int *nextLayer) {
256  int index = i * nCols + j;
257  int dir = dirMatrix[index];
258 
259  if ((dir & 1) && (j != nCols - 1)) {
260  if ((--inDegreeMatrix[index + 1]) == 0) {
261  nextLayer[numNextLayer++] = index + 1;
262  }
263  }
264 
265  if ((dir & 2) && i != nRows - 1 && j != nCols - 1) {
266  if ((--inDegreeMatrix[index + nCols + 1]) == 0) {
267  nextLayer[numNextLayer++] = index + nCols + 1;
268  }
269  }
270 
271  if ((dir & 4) && i != nRows - 1) {
272  if ((--inDegreeMatrix[index + nCols]) == 0) {
273  nextLayer[numNextLayer++] = index + nCols;
274  }
275  }
276 
277  if ((dir & 8) && i != nRows - 1 && j != 0) {
278  if ((--inDegreeMatrix[index + nCols - 1]) == 0) {
279  nextLayer[numNextLayer++] = index + nCols - 1;
280  }
281  }
282 
283  if ((dir & 16) && j != 0) {
284  if ((--inDegreeMatrix[index - 1]) == 0) {
285  nextLayer[numNextLayer++] = index - 1;
286  }
287  }
288 
289  if ((dir & 32) && i != 0 && j != 0) {
290  if ((--inDegreeMatrix[index - nCols - 1]) == 0) {
291  nextLayer[numNextLayer++] = index - nCols - 1;
292  }
293  }
294 
295  if ((dir & 64) && i != 0) {
296  if ((--inDegreeMatrix[index - nCols]) == 0) {
297  nextLayer[numNextLayer++] = index - nCols;
298  }
299  }
300 
301  if ((dir & 128) && i != 0 && j != nCols - 1) {
302  if ((--inDegreeMatrix[index - nCols + 1]) == 0) {
303  nextLayer[numNextLayer++] = index - nCols + 1;
304  }
305  }
306 }

308 string GridLayeringFromSource(int nValidGrids, const int *dirMatrix, const int *compressedIndex, int *inDegreeMatrix,
309  int nRows, int nCols, int dirNoDataValue, int outputNoDataValue, int *&layeringGrid) {
310  int n = nRows * nCols;
311  layeringGrid = new int[n]; //the value in this grid is the layering number
312 #pragma omp parallel for
313  for (int i = 0; i < n; i++) {
314  layeringGrid[i] = outputNoDataValue;
315  }
316 
317  int *lastLayer = new int[n];
318  int numLastLayer = 0; // the number of cells of last layer
319 
320  int index = 0;
321 
322  // 1. find source grids
323  for (int i = 0; i < nRows; i++) {
324  for (int j = 0; j < nCols; j++) {
325  index = i * nCols + j;
326 
327  if (dirMatrix[index] == dirNoDataValue || dirMatrix[index] < 0) {
328  continue;
329  } else {
330  if (inDegreeMatrix[index] == 0) {
331  lastLayer[numLastLayer++] = index;
332  }
333  }
334  }
335  }
336 
337  // 2. loop and layer
338  int numNextLayer = 0;
339  int curNum = 0; //the layering number of the current layer. In the end, it is the total number of layers.
340  int *nextLayer = new int[n];
341  int *tmp;
342  ostringstream oss;
343  while (numLastLayer > 0) {
344  oss << "\n" << numLastLayer << "\t";
345  curNum++;
346 
347  numNextLayer = 0;
348  for (int iInLayer = 0; iInLayer < numLastLayer; iInLayer++) {
349  index = lastLayer[iInLayer];
350  oss << compressedIndex[index] << "\t";
351 
352  layeringGrid[index] = curNum;
353 
354  int i = index / nCols;
355  int j = index % nCols;
356 
357  UpdateDownStream(i, j, nRows, nCols, dirMatrix, inDegreeMatrix, numNextLayer, nextLayer);
358  }
359 
360  numLastLayer = numNextLayer;
361 
362  tmp = lastLayer;
363  lastLayer = nextLayer;
364  nextLayer = tmp;
365  }
366 
367  delete[] lastLayer;
368  delete[] nextLayer;
369  lastLayer = NULL;
370  nextLayer = NULL;
371 
372  ostringstream outputOss;
373  outputOss << nValidGrids << "\t" << curNum << oss.str();
374 
375  return outputOss.str();
376 }

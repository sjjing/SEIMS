3 """Construct hillslope-slope position units.
4 
5  The main procedure:\n
6  1. Assign unique ID to each type of slope position unit with hillslopes, and
7  assign the up-down relationships according to the slope position sequence.\n
8  2. Statistics the landuse types and areas within each slope position units.\n
9  3. Merge hillslope with incomplete slope position sequences to other hillslopes
10  of the same subbasin.\n
11  4. Merge subbasin with incomplete slope position sequences to its downstream,
12  for the outlet subbasin merge to its upstream.
13 
14  @author : Liangjun Zhu, Huiran Gao
15  @changelog: 17-08-14 lj - initial implementation.\n

128  def assign_uniqueid_slppos_units(self):
129  """Get unique ID by multiply slope position value and hillslope ID"""
130  for m in range(self.nrows):
131  for n in range(self.ncols):
132  if self.data_slppos[m][n] == self.nodata_slppos or \
133  self.data_hillslp[m][n] == self.nodata_hillslp:
134  continue
135  cur_slppos = int(self.data_slppos[m][n])
136  cur_spname = self.slppos_tags.get(cur_slppos).get('name')
137  cur_hs = int(self.data_hillslp[m][n])
138  if cur_hs <= self.subbsin_num: # which is stream
139  continue
140  cur_subbsn = DelineateHillslope.get_subbasin_from_hillslope_id(cur_hs,
141  self.subbsin_num)
142  if cur_subbsn <= 0:
143  continue
144  if cur_slppos not in self.slppos_tags:
145  continue
146  cur_unit = cur_hs * cur_slppos
147  self.slppos_ids[m][n] = cur_unit
148  # calculate theoretical downslope and upslope unit ID (check later)
149  down_id = self.slppos_tags.get(cur_slppos).get('downslope') * cur_hs
150  up_id = self.slppos_tags.get(cur_slppos).get('upslope') * cur_hs
151  if down_id < 0:
152  down_id = -1
153  if up_id < 0:
154  up_id = -1
155  if cur_unit not in self.units_updwon.get(cur_spname):
156  self.units_updwon.get(cur_spname)[cur_unit] = {'area': 1,
157  'downslope': down_id,
158  'upslope': up_id,
159  'hillslope': [cur_hs],
160  'subbasin': [cur_subbsn],
161  'landuse': dict()}
162  else:
163  self.units_updwon.get(cur_spname)[cur_unit]['area'] += 1
164  # add landuse statistics
165  cur_lu = self.data_landuse[m][n]
166  if cur_lu != self.nodata_landuse and cur_lu > 0:
167  cur_lu = int(cur_lu)
168  if cur_lu not in self.units_updwon.get(cur_spname)[cur_unit]['landuse']:
169  self.units_updwon.get(cur_spname)[cur_unit]['landuse'][cur_lu] = 1
170  else:
171  self.units_updwon.get(cur_spname)[cur_unit]['landuse'][cur_lu] += 1

173  def check_unit_id(self):
174  """check the existence of upslope and downslope units."""
175  if 'overview' not in self.units_updwon:
176  self.units_updwon['overview'] = OrderedDict()
177  all_units_ids = list()
178  for sp in self.slppos_tags:
179  name = self.slppos_tags.get(sp).get('name')
180  cur_units_ids = self.units_updwon.get(name).keys()
181  self.units_updwon.get('overview')[name] = len(cur_units_ids)
182  all_units_ids += cur_units_ids
183  self.units_updwon.get('overview')['all_units'] = len(all_units_ids)
184 
185  for sp in self.slppos_tags:
186  name = self.slppos_tags.get(sp).get('name')
187  for unitid, unitdict in self.units_updwon.get(name).iteritems():
188  if unitdict.get('upslope') > 0 and unitdict.get('upslope') not in all_units_ids:
189  self.units_updwon.get(name).get(unitid)['upslope'] = -1
190  if unitdict.get('downslope') > 0 and unitdict.get('downslope') not in all_units_ids:
191  self.units_updwon.get(name).get(unitid)['downslope'] = -1
192  # calculate area
193  area_km2 = self.dx * self.dx * 1.e-6
194  self.units_updwon.get(name).get(unitid)['area'] *= area_km2
195  for luid in self.units_updwon.get(name).get(unitid)['landuse']:
196  self.units_updwon.get(name).get(unitid).get('landuse')[luid] *= area_km2

208  def merge_slopeposition_units(self):
209  """Merge hillslope/subbasin with incomplete slope position sequences"""
210  all_units_ids = list()
211  for sp in self.slppos_tags:
212  name = self.slppos_tags.get(sp).get('name')
213  all_units_ids += self.units_updwon.get(name).keys()
214  # loop each slope position units
215  hillslp_elim = dict()
216  for sp, spdict in self.slppos_tags.iteritems():
217  name = spdict.get('name')
218  for unitid, unitdict in self.units_updwon.get(name).iteritems():
219  flag = True # complete sequences or not
220  if spdict.get('upslope') > 0 > unitdict.get('upslope'):
221  flag = False
222  if spdict.get('downslope') > 0 > unitdict.get('downslope'):
223  flag = False
224  cur_hillslp_id = unitdict.get('hillslope')[0]
225  if not flag:
226  if cur_hillslp_id not in hillslp_elim:
227  hillslp_elim[cur_hillslp_id] = [unitid]
228  if cur_hillslp_id in hillslp_elim and unitid not in hillslp_elim[cur_hillslp_id]:
229  hillslp_elim[cur_hillslp_id].append(unitid)
230  print ('Hillslope with incomplete slope position sequences: ' + hillslp_elim.__str__())
231  # Basic procedure:
232  # 1. if header is incomplete, search left and right;
233  # if left/right is incomplete, search header and right/left;
234  # 2. if still not incomplete, check the subbasin,
235  # 3. if still not satisfied, merge the left and right to the left or right of downstream
236  # (or upstream for outlet subbasin). The header always with the left hillslope.
237  hillslope_merge_order = {0: [1, 2], 1: [0, 2], 2: [0, 1]}
238  units_pair = dict() # unit id pairs, key is merged unit, value is destination unit.
239  for hillid, unitids in hillslp_elim.iteritems():
240  subbsnid = DelineateHillslope.get_subbasin_from_hillslope_id(hillid, self.subbsin_num)
241  hillids = DelineateHillslope.cal_hs_codes(self.subbsin_num, subbsnid)
242  idx = hillids.index(hillid)
243  # slope positions that existed in current hillslope
244  sp_seq = list()
245  for _id in unitids:
246  _id /= hillid
247  if _id not in sp_seq:
248  sp_seq.append(_id)
249 
250  satisfied = False
251  cur_units_pair = dict()
252  for curidx in hillslope_merge_order[idx]:
253  dst_hillid = hillids[curidx]
254  for _id in sp_seq:
255  dst_unit = _id * dst_hillid
256  if dst_unit in all_units_ids:
257  cur_units_pair[_id * hillid] = dst_unit
258  for sp in self.slppos_tags:
259  _uid = sp * dst_hillid
260  if _uid in all_units_ids and sp not in sp_seq:
261  sp_seq.append(sp)
262  if SlopePositionUnits.check_slppos_sequence(sp_seq, self.slppos_tags):
263  satisfied = True
264  break
265  if satisfied:
266  units_pair.update(cur_units_pair)
267  else: # means the current subbasin should be merged to downstream
268  downstream_merge_idx = {0: 1, 1: 1, 2: 2}
269  cur_units_pair = dict()
270  dst_subbsn = self.subbsin_tree.get(subbsnid)
271  if dst_subbsn < 0: # outlet subbasin
272  for k, v in self.subbsin_tree.iteritems():
273  if v == subbsnid:
274  dst_subbsn = k
275  break
276  if dst_subbsn > 0: # although dst_subbsn may not less than 0, just make sure that!
277  dst_hillids = DelineateHillslope.cal_hs_codes(self.subbsin_num, dst_subbsn)
278  dst_downstream_hs = dst_hillids[downstream_merge_idx[idx]]
279  for _id in sp_seq:
280  dst_unit = _id * dst_downstream_hs
281  if dst_unit in all_units_ids:
282  cur_units_pair[_id * hillid] = dst_unit
283  units_pair.update(cur_units_pair)
284 
285  print ('Slope position unit merge-destination pairs: ' + units_pair.__str__())
286 
287  # begin to merge
288  for srcid, dstid in units_pair.iteritems():
289  # replace values in slope position units raster
290  for k, v in units_pair.iteritems():
291  self.slppos_ids[self.slppos_ids == k] = v
292 
293  for sp in self.slppos_tags:
294  name = self.slppos_tags.get(sp).get('name')
295  if srcid in self.units_updwon.get(name).keys(): # dstid MUST be here too.
296  srcdict = self.units_updwon.get(name)[srcid]
297  self.units_updwon[name][dstid]['area'] += srcdict['area']
298  for luid in srcdict['landuse']:
299  if luid not in self.units_updwon.get(name)[dstid]['landuse']:
300  self.units_updwon[name][dstid]['landuse'][luid] = \
301  srcdict['landuse'][luid]
302  else:
303  self.units_updwon[name][dstid]['landuse'][luid] += \
304  srcdict['landuse'][luid]
305  if srcdict['hillslope'][0] not in self.units_updwon[name][dstid]['hillslope']:
306  self.units_updwon[name][dstid]['hillslope'].append(srcdict['hillslope'][0])
307  if srcdict['subbasin'][0] not in self.units_updwon[name][dstid]['subbasin']:
308  self.units_updwon[name][dstid]['subbasin'].append(srcdict['subbasin'][0])
309  self.units_updwon.get(name).pop(srcid)
310  break
311 
312  # rewrite the overview
313  units_count = 0
314  for sp in self.slppos_tags:
315  name = self.slppos_tags.get(sp).get('name')
316  cur_units_count = len(self.units_updwon.get(name).keys())
317  self.units_updwon.get('overview')[name] = cur_units_count
318  units_count += cur_units_count
319  self.units_updwon.get('overview')['all_units'] = units_count

321  def extract_subbasin_hillslope_slppos(self):
322  """Extract the hierarchical relationship of spatial units, i.e.,

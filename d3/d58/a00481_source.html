34 int PER_STR::Execute() {
35  CheckInputData();
36  initialOutputs();
37 #pragma omp parallel for
38  for (int i = 0; i < m_nCells; i++) {
39  // Note that, infiltration, pothole seepage, irrigation etc. have been added to
40  // the first soil layer in other modules. By LJ
41 
42  float excessWater = 0.f, maxSoilWater = 0.f, fcSoilWater = 0.f;
43  for (int j = 0; j < (int) m_soilLayers[i]; j++) {
44  excessWater = 0.f;
45  maxSoilWater = m_sat[i][j];
46  fcSoilWater = m_fc[i][j];
47  // determine gravity drained water in layer
48  excessWater += m_soilStorage[i][j] - fcSoilWater;
49  //if (i == 100)
50  // cout<<"lyr: "<<j<<", soil storage: "<<m_soilStorage[i][j]<<", fc: "<<fcSoilWater<<", excess: "<<excessWater<<endl;
51  // for the upper two layers, soil may be frozen
52  if (j == 0 && m_soilT[i] <= m_frozenT) {
53  continue;
54  }
55  m_perc[i][j] = 0.f;
56  // No movement if soil moisture is below field capacity
57  if (excessWater > 1.e-5f) {
58  float maxPerc = maxSoilWater - fcSoilWater;
59  float tt = 3600.f * maxPerc / m_ks[i][j]; // secs
60  m_perc[i][j] = excessWater * (1.f - exp(-m_dt / tt)); // secs
61 
62  if (m_perc[i][j] > maxPerc) {
63  m_perc[i][j] = maxPerc;
64  }
65  //Adjust the moisture content in the current layer, and the layer immediately below it
66  m_soilStorage[i][j] -= m_perc[i][j];
67  excessWater -= m_perc[i][j];
68  m_soilStorage[i][j] = max(UTIL_ZERO, m_soilStorage[i][j]);
69  // redistribute soil water if above field capacity (high water table), rewrite from sat_excess.f of SWAT
70  //float qlyr = m_soilStorage[i][j];
71  if (j < (int) m_soilLayers[i] - 1) {
72  m_soilStorage[i][j + 1] += m_perc[i][j];
73  if (m_soilStorage[i][j] - m_sat[i][j] > 1.e-4f) {
74  m_soilStorage[i][j + 1] += m_soilStorage[i][j] - m_sat[i][j];
75  m_soilStorage[i][j] = m_sat[i][j];
76  }
77  } else
78  {
79  if (m_soilStorage[i][j] - m_sat[i][j] > 1.e-4f) {
80  float ul_excess = m_soilStorage[i][j] - m_sat[i][j];
81  m_soilStorage[i][j] = m_sat[i][j];
82  for (int ly = (int) m_soilLayers[i] - 2; ly >= 0; ly--) {
83  m_soilStorage[i][ly] += ul_excess;
84  if (m_soilStorage[i][ly] > m_sat[i][ly]) {
85  ul_excess = m_soilStorage[i][ly] - m_sat[i][ly];
86  m_soilStorage[i][ly] = m_sat[i][ly];
87  } else {
88  ul_excess = 0.f;
89  break;
90  }
91  if (ly == 0 && ul_excess > 0.f) {
92  // add ul_excess to depressional storage and then to surfq
93  if (m_potVol != NULL) {
94  m_potVol[i] += ul_excess;
95  } else {
96  m_surfQmm[i] += ul_excess;
97  }
98  }
99  }
100  }
101  }
102  } else {
103  m_perc[i][j] = 0.f;
104  }
105  }
107  m_soilStorageProfile[i] = 0.f;
108  for (int ly = 0; ly < (int) m_soilLayers[i]; ly++) {
109  m_soilStorageProfile[i] += m_soilStorage[i][ly];
110  }
111  }
112  return 0;
113 }

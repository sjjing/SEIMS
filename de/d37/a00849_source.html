43  """Algorithm modified from Whitebox GAT v3.4.0.
44  source code: https://github.com/jblindsay/whitebox-geospatial-analysis-tools/
45  blob/master/HydroTools/src/plugins/Hillslopes.java
46  Args:
47  stream_raster: Stream cell value greater than 0 is identified by stream
48  The input stream are recommended sequenced as 1, 2, 3...
49  flow_dir_raster: D8 flow direction in TauDEM code
50  hillslope_out: With the sequenced stream IDs, the output hillslope will be numbered:
51  - Header hillslope: MaxStreamID + (current_id - 1) * 3 + 1
52  - Right hillslope: MaxStreamID + (current_id - 1) * 3 + 2
53  - Left hillslope: MaxStreamID + (current_id - 1) * 3 + 3
54  d8alg: Currently, "TauDEM", "ArcGIS", and "Whitebox" are supported.
55  stream_value_method: stream value assigned method, depend on this parameter,
56  the output hillslope will be appended as follows:
57  -1 - all the four files will be output.
58  0 - keep stream link code, which has the default file name
59  1 - Set to the value of right hillslope and head hillslope, <name>_r.tif
60  2 - Set to the value of left hillslope and head hillslope, <name>_l.tif
61  3 - Set stream cell to NoData, <name>_n.tif
62  """

82  def inflow_stream_number(vrow, vcol, flowmodel="taudem"):
83  """
84  Count the inflow stream cell number and coordinates of all inflow cells
85  Args:
86  vrow: row number
87  vcol: col number
88  flowmodel: D8 flow direction algorithm.
89  Returns:
90  neighb_stream_cell_num: inflow cells number that is stream
91  cell_coors: inflow cell coordinates, the size() is equal or greater
92  than neighb_stream_cell_num
93  """
94  neighb_stream_cell_num = 0
95  cell_coors = []
96  for c in range(8):
97  newrow = vrow + FlowModelConst.ccw_drow[c]
98  newcol = vcol + FlowModelConst.ccw_dcol[c]
99  if newrow < 0 or newrow >= nrows or newcol < 0 or newcol >= ncols:
100  continue
101  if flowd8_data[newrow][newcol] == FlowModelConst.d8_inflows.get(flowmodel)[c]:
102  cell_coors.append((newrow, newcol))
103  if stream_data[newrow][newcol] > 0 \
104  and stream_data[newrow][newcol] != stream_nodata:
105  neighb_stream_cell_num += 1
106  return neighb_stream_cell_num, cell_coors

108  def assign_sequenced_stream_ids(c_id, vrow, vcol, flowmodel="taudem"):
109  """set sequenced stream IDs"""
110  in_strm_num, in_coors = inflow_stream_number(vrow, vcol, flowmodel)
111  if in_strm_num == 0:
112  # it's a headwater location so start a downstream flowpath
113  c_id += 1
114  tmp_row = vrow
115  tmp_col = vcol
116  sequenced_stream_d[tmp_row][tmp_col] = c_id
117  searched_flag = True
118  while searched_flag:
119  # find the downslope neighbour
120  tmpflowd8 = flowd8_data[tmp_row][tmp_col]
121  if tmpflowd8 < 0 or tmpflowd8 == flowd8_nodata:
122  if stream_data[tmp_row][tmp_col] > 0 \
123  and stream_data[tmp_row][tmp_col] != stream_nodata:
124  # it is a valid stream cell and probably just has no downslope
125  # neighbour (e.g. at the edge of the grid)
126  sequenced_stream_d[tmp_row][tmp_col] = c_id
127  break
128  tmp_row, tmp_col = D8Util.downstream_index(tmpflowd8, tmp_row,
129  tmp_col, flowmodel)
130  if tmp_row < 0 or tmp_row >= nrows or tmp_col < 0 or tmp_col >= ncols:
131  break
132  if stream_data[tmp_row][tmp_col] <= 0:
133  searched_flag = False # it is not a stream cell
134  else:
135  if sequenced_stream_d[tmp_row][tmp_col] > 0:
136  # run into a larger stream, end the downstream search
137  break
138  # is it a confluence (conjunction node)
139  in_strm_num, in_coors = inflow_stream_number(tmp_row, tmp_col, flowmodel)
140  if in_strm_num >= 2:
141  c_id += 1
142  sequenced_stream_d[tmp_row][tmp_col] = c_id
143  return c_id

145  def assign_hillslope_code_of_neighbors(vrow, vcol, flowmodel="taudem"):
146  """set hillslope code for neighbors of current stream cell."""
147  stream_coors.append((vrow, vcol))
148  in_strm_num, in_coors = inflow_stream_number(vrow, vcol, flowmodel)
149  strm_id = stream_data[vrow][vcol]
150  # print ("Assign hillslope code for stream cell, r: %d, c: %d, ID: %d" % (vrow, vcol,
151  # int(strm_id)))
152  # set hillslope IDs
153  hillslp_ids = DelineateHillslope.cal_hs_codes(max_id, strm_id)
154  cur_d8_value = flowd8_data[vrow][vcol]
155  if in_strm_num == 0: # it is a one-order stream head
156  headstream_coors.append((vrow, vcol))
157  for (in_nostrm_row, in_nostrm_col) in in_coors:
158  hillslope_mtx[in_nostrm_row][in_nostrm_col] = hillslp_ids[0]
159  else: # search the 3*3 neighbors by clockwise and counterclockwise separately
160  if cur_d8_value <= 0 or cur_d8_value == flowd8_nodata:
161  return
162  dirv = int(cur_d8_value) # direction code
163  d_idx = FlowModelConst.d8_dirs.get(flowmodel).index(dirv) # direction index
164  # look to the right side, i.e. clockwise
165  d_idx_r = d_idx
166  while True and len(in_coors) > 0:
167  d_idx_r -= 1
168  if d_idx_r > 7:
169  d_idx_r = 0
170  if d_idx_r < 0:
171  d_idx_r = 7
172  tmp_row = vrow + FlowModelConst.ccw_drow[d_idx_r]
173  tmp_col = vcol + FlowModelConst.ccw_dcol[d_idx_r]
174  if (tmp_row, tmp_col) not in in_coors: # not inflow to this cell
175  continue
176  tmpstream = stream_data[tmp_row][tmp_col]
177  in_coors.remove((tmp_row, tmp_col))
178  if tmpstream <= 0 or tmpstream == stream_nodata:
179  hillslope_mtx[tmp_row][tmp_col] = hillslp_ids[1] # right hillslope
180  else: # encounter another in flow stream
181  break
182 
183  # look to the left side, i.e. counterclockwise
184  d_idx_l = d_idx
185  while True and len(in_coors) > 0:
186  d_idx_l += 1
187  if d_idx_l > 7:
188  d_idx_l = 0
189  if d_idx_l < 0:
190  d_idx_l = 7
191  tmp_row = vrow + FlowModelConst.ccw_drow[d_idx_l]
192  tmp_col = vcol + FlowModelConst.ccw_dcol[d_idx_l]
193  if (tmp_row, tmp_col) not in in_coors: # not inflow to this cell
194  continue
195  tmpstream = stream_data[tmp_row][tmp_col]
196  in_coors.remove((tmp_row, tmp_col))
197  if tmpstream <= 0 or tmpstream == stream_nodata:
198  hillslope_mtx[tmp_row][tmp_col] = hillslp_ids[2] # left hillslope
199  else: # encounter another in flow stream
200  break
201  # if any inflow cells existed?
202  if len(in_coors) > 0:
203  for (tmp_row, tmp_col) in in_coors:
204  tmpstream = stream_data[tmp_row][tmp_col]
205  if tmpstream <= 0 or tmpstream == stream_nodata:
206  hillslope_mtx[tmp_row][tmp_col] = hillslp_ids[0]
207  # add the current cell as head stream
208  headstream_coors.append((vrow, vcol))

91 int NPS_Management::Execute() {
92  CheckInputData();
93  if (m_cellArea < 0.f) m_cellArea = m_cellWidth * m_cellWidth;
94  for (map<int, BMPArealSrcFactory *>::iterator it = m_arealSrcFactory.begin(); it != m_arealSrcFactory.end(); it++) {
96  if (!it->second->GetLocationLoadStatus()) {
97  it->second->SetArealSrcLocsMap(m_nCells, m_mgtFields);
98  }
100  vector<int> tmpArealFieldIDs = it->second->GetArealSrcIDs();
101  vector<int> tmpMgtSeqs = it->second->GetArealSrcMgtSeqs();
102  for (vector<int>::iterator mgtSeqIter = tmpMgtSeqs.begin(); mgtSeqIter != tmpMgtSeqs.end(); mgtSeqIter++) {
103  ArealSourceMgtParams *tmpMgtParams = it->second->GetArealSrcMgtMap().at(*mgtSeqIter);
104  if (tmpMgtParams->GetStartDate() != 0 && tmpMgtParams->GetEndDate() != 0) {
105  if (m_date < tmpMgtParams->GetStartDate() || m_date > tmpMgtParams->GetEndDate()) {
106  continue;
107  }
108  }
109  for (vector<int>::iterator fIDIter = tmpArealFieldIDs.begin(); fIDIter != tmpArealFieldIDs.end();
110  fIDIter++) {
111  float deltaWtrMM = 0.f, deltaNH4 = 0.f, deltaNO3 = 0.f, deltaOrgN = 0.f;
112  float deltaMinP = 0.f, deltaOrgP = 0.f;
113  map < int, ArealSourceLocations * > tmpLocMap = it->second->GetArealSrcLocsMap();
114  if (tmpLocMap.find(*fIDIter) == tmpLocMap.end()) {
115  continue;
116  }
117  ArealSourceLocations *tmpLoc = tmpLocMap.at(*fIDIter);
118  float tmpSize = tmpLoc->GetSize();
119  int tmpNCells = tmpLoc->GetValidCells();
120  if (tmpMgtParams->GetWaterVolume() > 0.f) {
121  deltaWtrMM =
122  tmpMgtParams->GetWaterVolume() * tmpSize * m_timestep / 86400.f / tmpNCells / m_cellArea *
123  1000.f;
124  }
125  float cvt =
126  tmpSize * m_timestep / 86400.f / tmpNCells / m_cellArea * 10000.f;
127  if (tmpMgtParams->GetNH4() > 0.f) {
128  deltaNH4 = cvt * tmpMgtParams->GetNH4();
129  }
130  if (tmpMgtParams->GetNO3() > 0.f) {
131  deltaNO3 = cvt * tmpMgtParams->GetNO3();
132  }
133  if (tmpMgtParams->GetOrgN() > 0.f) {
134  deltaOrgN = cvt * tmpMgtParams->GetOrgN();
135  }
136  if (tmpMgtParams->GetMinP() > 0.f) {
137  deltaMinP = cvt * tmpMgtParams->GetMinP();
138  }
139  if (tmpMgtParams->GetOrgP() > 0.f) {
140  deltaOrgP = cvt * tmpMgtParams->GetOrgP();
141  }
142  vector<int> tmpCellIdx = tmpLoc->GetCellsIndex();
143  for (vector<int>::iterator idxIter = tmpCellIdx.begin(); idxIter != tmpCellIdx.end(); idxIter++) {
145  if (deltaWtrMM > 0.f && m_soilStorage != NULL) {
146  m_soilStorage[*idxIter][0] += deltaWtrMM;
147  }
148  if (deltaNO3 > 0.f && m_sol_no3 != NULL) {
149  m_sol_no3[*idxIter][0] += deltaNO3;
150  }
151  if (deltaNH4 > 0.f && m_sol_nh4 != NULL) {
152  m_sol_nh4[*idxIter][0] += deltaNH4;
153  }
154  if (deltaMinP > 0.f && m_sol_solp != NULL) {
155  m_sol_solp[*idxIter][0] += deltaMinP;
156  }
157  if (deltaOrgN > 0.f && m_sol_orgn != NULL) {
158  m_sol_orgn[*idxIter][0] += deltaOrgN;
159  }
160  if (deltaOrgP > 0.f && m_sol_orgp != NULL) {
161  m_sol_orgp[*idxIter][0] += deltaOrgP;
162  }
163  }
164  }
165  }
166  }

168  //cout<<", new: "<<m_sol_solp[46364][0]<<endl;
169  return true;
170 }

24 class SoilProperty(object):
25  """
26  base class of Soil physical and general chemical properties
27  Attributes:
28  SOILLAYERS (int, None): (nly in SWAT model, the same below) number of soil layers
29  SOILDEPTH (float, mm): (sol_z) depth from the surface to bottom of soil layer
30  SOILTHICK (float, mm): soil thickness for calculation convenient, derived from `SOILDEPTH`
31  OM (float, %): organic matter content (weight percent)
32  SOL_CBN (float, %): (sol_cbn) percent organic carbon in soil layer, calculated by `OM`
33  SOL_N (float, %): (sol_n) used when using CSWAT = 1, i.e, C-FARM one carbon pool model, derived from `SOL_CBN`
34  CLAY (float, %): (sol_clay) percent clay content in soil material, diameter < 0.002 mm
35  SILT (float, %): (sol_silt) percent silt content in soil material,diameter between 0.002 mm and 0.05 mm
36  SAND (float, %): (sol_sand) percent sand content in soil material,diameter between 0.05 mm and 2 mm
37  ROCK (float, %): (sol_rock) percent of rock fragments content in soil material,diameter > 2 mm
38  SOIL_TEXTURE (int, None): soil texture code used in WetSpa and SWAT model based on soil texture triangle by USDA
39  HYDRO_GROUP (int, None): Hydrological soil group, 1, 2, 3, and 4 to represent A, B, C, and D
40  SOL_ZMX (float, mm): (sol_zmx) maximum rooting depth of soil profile
41  ANION_EXCL (float, None): (anion_excl) fraction of porosity from which anions are excluded, default is 0.5
42  SOL_CRK (float, None): (sol_crk) crack volume potential of soil expressed as a fraction of the total soil volume
43  DENSITY (float, Mg/m3 or g/cm3): (sol_bd) bulk density of each soil layer
44  SOL_AVBD (float, Mg/m3 or g/cm3): (sol_avbd) average bulk density for soil profile
45  CONDUCTIVITY (float, mm/hr): (sol_k) saturated hydraulic conductivity
46  SOL_HK (float, None): (sol_hk) beta coefficent to calculate hydraulic conductivity
47  WILTINGPOINT (float, mm H2O / mm soil): (sol_wp) water content of soil at -1.5 MPa (wilting point)
48  SOL_WPMM (float, mm H2O): (sol_wpmm) water content of soil at -1.5 MPa (wilting point)
49  SOL_SUMWP (float, mm H2O): (sol_sumwp) amount of water held in the soil profile at wilting point
50  FIELDCAP (float,mm H2O / mm soil): (sol_up) water content of soil at -0.033 MPa (field capacity)
51  AWC (float,mm H2O / mm soil): (sol_awc) available water capacity of soil layer
52  SOL_AWC (float,mm H2O): (sol_fc) amount of water available to plants in soil layer at field capacity (fc - wp)
53  SOL_SUMAWC (float,mm H2O): (sol_sumfc) amount of water held in soil profile at field capacity
54  POROSITY (float,None): (sol_por) total porosity of soil layer expressed as a fraction of the total volume
55  POREINDEX (float,None): pore size distribution index
56  SOL_AVPOR (float,None): (sol_avpor) average porosity for entire soil profile
57  SOL_UL (float,mm H2O): (sol_ul) amount of water held in the soil layer at saturation (sat - wp water)
58  SOL_SUMUL (float,mm H2O): (sol_sumul) amount of water held in soil profile at saturation
59  USLE_K (float,None): USLE K factor
60  SOL_ALB (float,None): albedo of top soil surface
61  WFSH (float,mm): wetting front matric potential (usde in Green-Ampt method)
62  ESCO (float,None): soil evaporation compensation factor
63  VWT (float,None): (vwt) variable water table factor, used in percolation modules
64  DET_SAND (float,None): (det_san) detached sediment size distribution, sand fraction
65  DET_SILT (float,None): (det_sil) detached sediment size distribution, silt fraction
66  DET_CLAY (float,None): (det_cla) detached sediment size distribution, clay fraction
67  DET_SMAGG (float,None): (det_sag) detached sediment size distribution, small aggregation fraction
68  DET_LGAGG (float,None): (det_lag) detached sediment size distribution, large aggregation fraction
69  CRDEP (float,mm): (crdep) maximum or potential crack volume
70  VOLCR (float,mm): (volcr) crack volume for soil layer, should be calculated in SEIMS, using moist_ini
71  SOL_NO3 (float,kg/ha): (sol_no3) concentration of nitrate in soil layers
72  SOL_NH4 (float,kg/ha): (sol_nh4) concentration of ammonium-N in soil layers
73  SOL_ORGN (float,kg/ha): (sol_orgn) organic N concentration in soil layers
74  SOL_ORGP (float,kg/ha): (sol_orgp) organic P concentration in soil layers
75  SOL_SOLP (float,kg/ha): (sol_solp) soluble P concentration in soil layers
76  """

202  if self.SOL_NO3:
203  self.SOL_NO3.insert(0, self.SOL_NO3[0])
204  else:
205  self.SOL_NO3 = list(numpy.zeros(self.SOILLAYERS))
206  if self.SOL_NH4:
207  self.SOL_NH4.insert(0, self.SOL_NH4[0])
208  else:
209  self.SOL_NH4 = list(numpy.zeros(self.SOILLAYERS))
210  if self.SOL_ORGN:
211  self.SOL_ORGN.insert(0, self.SOL_ORGN[0])
212  else:
213  self.SOL_ORGN = list(numpy.zeros(self.SOILLAYERS))
214  if self.SOL_SOLP:
215  self.SOL_SOLP.insert(0, self.SOL_SOLP[0])
216  else:
217  self.SOL_SOLP = list(numpy.zeros(self.SOILLAYERS))
218  if self.SOL_ORGP:
219  self.SOL_ORGP.insert(0, self.SOL_ORGP[0])
220  else:
221  self.SOL_ORGP = list(numpy.zeros(self.SOILLAYERS))
222  if self.SOILDEPTH == [] or len(self.SOILDEPTH) != self.SOILLAYERS or DEFAULT_NODATA in self.SOILDEPTH:
223  raise IndexError("Soil depth must have a size equal to NLAYERS and should not include NODATA (-9999)!")
224  # Calculate soil thickness of each layer
225  for l in range(self.SOILLAYERS):
226  if l == 0:
227  self.SOILTHICK.append(self.SOILDEPTH[l])
228  else:
229  self.SOILTHICK.append(self.SOILDEPTH[l] - self.SOILDEPTH[l - 1])
230  if self.SOL_ZMX == DEFAULT_NODATA or self.SOL_ZMX > self.SOILDEPTH[-1]:
231  self.SOL_ZMX = self.SOILDEPTH[-1]
232  if self.ANION_EXCL == DEFAULT_NODATA:
233  self.ANION_EXCL = 0.5
234  if not self.OM or len(self.OM) != self.SOILLAYERS:
235  raise IndexError("Soil organic matter must have a size equal to NLAYERS!")
236  elif DEFAULT_NODATA in self.OM and self.OM.index(DEFAULT_NODATA) >= 2 and self.SOILLAYERS >= 3:
237  for i in range(2, self.SOILLAYERS):
238  if self.OM[i] == DEFAULT_NODATA:
239  self.OM[i] = self.OM[i - 1] * numpy.exp(-self.SOILTHICK[i]) # mm
240  # Calculate sol_cbn = sol_om * 0.58
241  if not self.SOL_CBN or len(self.SOL_CBN) != self.SOILLAYERS:
242  self.SOL_CBN = []
243  for i in range(self.SOILLAYERS):
244  if self.OM[i] * 0.58 < UTIL_ZERO:
245  self.SOL_CBN.append(0.1)
246  else:
247  self.SOL_CBN.append(self.OM[i] * 0.58)
248  # Calculate sol_n = sol_cbn/11.
249  if not self.SOL_N or len(self.SOL_N) != self.SOILLAYERS:
250  self.SOL_N = []
251  for i in range(self.SOILLAYERS):
252  self.SOL_N.append(self.SOL_CBN[i] / 11.)
253  if not self.CLAY or len(self.CLAY) != self.SOILLAYERS or DEFAULT_NODATA in self.CLAY:
254  raise IndexError("Clay content must have a size equal to NLAYERS and should not include NODATA (-9999)!")
255  if not self.SILT or len(self.SILT) != self.SOILLAYERS or DEFAULT_NODATA in self.SILT:
256  raise IndexError("Silt content must have a size equal to NLAYERS and should not include NODATA (-9999)!")
257  if not self.SAND or len(self.SAND) != self.SOILLAYERS or DEFAULT_NODATA in self.SAND:
258  raise IndexError("Sand content must have a size equal to NLAYERS and should not include NODATA (-9999)!")
259  if not self.ROCK or len(self.ROCK) != self.SOILLAYERS or DEFAULT_NODATA in self.ROCK:
260  raise IndexError("Rock content must have a size equal to NLAYERS and should not include NODATA (-9999)!")
261 
262  # temperory variables
263  tmp_fc = []
264  tmp_sat = []
265  tmp_wp = []
266  for i in range(self.SOILLAYERS):
267  s = self.SAND[i] * 0.01 # % -> decimal
268  c = self.CLAY[i] * 0.01
269  om = self.OM[i]
270  wpt = -0.024 * s + 0.487 * c + 0.006 * om + 0.005 * s * om - 0.013 * c * om + 0.068 * s * c + 0.031
271  tmp_wp.append(1.14 * wpt - 0.02)
272  fct = -0.251 * s + 0.195 * c + 0.011 * om + 0.006 * s * om - 0.027 * c * om + 0.452 * s * c + 0.299
273  fc = fct + 1.283 * fct * fct - 0.374 * fct - 0.015
274  s33t = 0.278 * s + 0.034 * c + 0.022 * om - 0.018 * s * om - 0.027 * c * om - 0.584 * s * c + 0.078
275  s33 = 1.636 * s33t - 0.107
276  sat = fc + s33 - 0.097 * s + 0.043
277  tmp_fc.append(fc)
278  tmp_sat.append(sat)
279 
280  if self.WILTINGPOINT and len(self.WILTINGPOINT) != self.SOILLAYERS:
281  raise IndexError(
282  "Wilting point must have a size equal to soil layers number!")
283  elif not self.WILTINGPOINT:
284  self.WILTINGPOINT = tmp_wp[:]
285  elif DEFAULT_NODATA in self.WILTINGPOINT:
286  for i in range(self.SOILLAYERS):
287  if self.WILTINGPOINT[i] == DEFAULT_NODATA:
288  self.WILTINGPOINT[i] = tmp_wp[i]
289  if self.DENSITY and len(self.DENSITY) != self.SOILLAYERS:
290  raise IndexError(
291  "Bulk density must have a size equal to soil layers number!")
292  elif not self.DENSITY or DEFAULT_NODATA in self.DENSITY:
293  tmp_bd = []
294  for i in range(self.SOILLAYERS):
295  sat = tmp_sat[i]
296  fc = tmp_fc[i]
297  if self.FIELDCAP and len(self.FIELDCAP) == self.SOILLAYERS:
298  sat = sat - fc + self.FIELDCAP[i]
299  tmp_bd.append(2.65 * (1.0 - sat))
300  if DEFAULT_NODATA in self.DENSITY:
301  for i in range(self.SOILLAYERS):
302  if self.DENSITY[i] == DEFAULT_NODATA:
303  self.DENSITY[i] = tmp_bd[i]
304  elif not self.DENSITY:
305  self.DENSITY = tmp_bd[:]
306  if self.FIELDCAP and len(self.FIELDCAP) != self.SOILLAYERS:
307  raise IndexError(
308  "Field capacity must have a size equal to soil layers number!")
309  elif not self.FIELDCAP or DEFAULT_NODATA in self.FIELDCAP:
310  tmp_fc_bdeffect = []
311  for i in range(self.SOILLAYERS):
312  fc = tmp_fc[i]
313  sat = tmp_sat[i]
314  if self.DENSITY != [] and len(self.DENSITY) == self.SOILLAYERS:
315  p_df = self.DENSITY[i]
316  else:
317  p_df = 2.65 * (1.0 - sat)
318  sat_df = 1. - p_df / 2.65
319  tmp_fc_bdeffect.append(fc - 0.2 * (sat - sat_df))
320  if DEFAULT_NODATA in self.FIELDCAP:
321  for i in range(self.SOILLAYERS):
322  if self.FIELDCAP[i] == DEFAULT_NODATA:
323  self.FIELDCAP[i] = tmp_fc_bdeffect[i]
324  elif not self.FIELDCAP:
325  self.FIELDCAP = tmp_fc_bdeffect[:]
326  if self.AWC and len(self.AWC) != self.SOILLAYERS:
327  raise IndexError("Available water capacity must have the size equal to soil layers number!")
328  elif not self.AWC:
329  for i in range(self.SOILLAYERS):
330  self.AWC.append(self.FIELDCAP[i] - self.WILTINGPOINT[i])
331  elif DEFAULT_NODATA in self.AWC:
332  for i in range(self.SOILLAYERS):
333  if self.AWC[i] == DEFAULT_NODATA:
334  self.AWC[i] = self.FIELDCAP[i] - self.WILTINGPOINT[i]
335 
336  if self.POREINDEX and len(self.POREINDEX) != self.SOILLAYERS:
337  raise IndexError("Pore disconnectedness index must have a size equal to soil layers number!")
338  elif not self.POREINDEX:
339  for i in range(self.SOILLAYERS):
340  # An fitted equation proposed by Cosby et al. (1984) is adopted. By LJ, 2016-9-22
341  b = 0.159 * self.CLAY[i] + 2.91
342  self.POREINDEX.append(b)
343  # previous version, currently deprecated by LJ
344  # fc = self.FIELDCAP[i]
345  # wp = self.WILTINGPOINT[i]
346  # b = (math.log(1500.) - math.log(33.)) / (math.log(fc) - math.log(wp))
347  # self.POREINDEX.append(1.0 / b)
348  if self.POROSITY and len(self.POROSITY) != self.SOILLAYERS:
349  raise IndexError(
350  "Soil Porosity must have a size equal to soil layers number!")
351  elif not self.POROSITY:
352  for i in range(self.SOILLAYERS):
353  self.POROSITY.append(1. - self.DENSITY[i] / 2.65) # from the theory of swat
354  elif DEFAULT_NODATA in self.POROSITY:
355  for i in range(self.SOILLAYERS):
356  if self.POROSITY[i] == DEFAULT_NODATA:
357  self.POROSITY[i] = 1. - self.DENSITY[i] / 2.65
358  if self.CONDUCTIVITY and len(self.CONDUCTIVITY) != self.SOILLAYERS:
359  raise IndexError("Saturated conductivity must have a size equal to soil layers number!")
360  elif not self.CONDUCTIVITY or DEFAULT_NODATA in self.CONDUCTIVITY:
361  tmp_k = []
362  for i in range(self.SOILLAYERS):
363  lamda = self.POREINDEX[i]
364  fc = tmp_fc[i]
365  sat = tmp_sat[i]
366  tmp_k.append(1930. * pow(sat - fc, 3. - lamda))
367  if not self.CONDUCTIVITY:
368  self.CONDUCTIVITY = tmp_k[:]
369  elif DEFAULT_NODATA in self.CONDUCTIVITY:
370  for i in range(self.SOILLAYERS):
371  if self.CONDUCTIVITY[i] == DEFAULT_NODATA:
372  self.CONDUCTIVITY[i] = tmp_k[i]
373  # calculate water content of soil at -1.5 MPa and -0.033 MPa, refers to soil_phys.f in SWAT
374  if not self.WILTINGPOINT:
375  for i in range(self.SOILLAYERS):
376  tmpwp = 0.4 * self.CLAY[i] * self.DENSITY[i] / 100.
377  if tmpwp <= 0.:
378  tmpwp = 0.005
379  self.WILTINGPOINT.append(tmpwp)
380  # calculate field capcity (sol_up)
381  if not self.FIELDCAP:
382  for i in range(self.SOILLAYERS):
383  self.FIELDCAP.append(self.WILTINGPOINT[i] + self.AWC[i])
384  # calculate porosity
385  if not self.POROSITY:
386  for i in range(self.SOILLAYERS):
387  self.POROSITY.append(1. - self.DENSITY[i] / 2.65)
388  if self.SOL_CRK == DEFAULT_NODATA:
389  self.SOL_CRK = numpy.mean(self.POROSITY)
390  for i in range(self.SOILLAYERS):
391  if self.FIELDCAP[i] >= self.POROSITY[i]:
392  self.FIELDCAP[i] = self.POROSITY[i] - 0.05
393  self.WILTINGPOINT[i] = self.FIELDCAP[i] - self.AWC[i]
394  if self.WILTINGPOINT[i] <= 0.:
395  self.FIELDCAP[i] = self.POROSITY[i] * 0.75
396  self.WILTINGPOINT[i] = self.POROSITY[i] * 0.25
397  # compute drainable porosity and variable water table factor
398  drpor = self.POROSITY[i] - self.FIELDCAP[i]
399  self.VWT.append((437.13 * drpor * drpor) - (95.08 * drpor) + 8.257)
400  sa = self.SAND[0] / 100.
401  cl = self.CLAY[0] / 100.
402  si = self.SILT[0] / 100.
403  # determine detached sediment size distribution typical for mid-western soils in USA (Foster et al., 1980)
404  # Based on SWRRB.
405  self.DET_SAND = 2.49 * sa * (1. - cl)
406  self.DET_SILT = 0.13 * si
407  self.DET_CLAY = 0.2 * cl
408  if cl < 0.25:
409  self.DET_SMAGG = 2.0 * cl
410  elif cl > 0.5:
411  self.DET_SMAGG = 0.57
412  else:
413  self.DET_SMAGG = 0.28 * (cl - 0.25) + 0.5
414  self.DET_LGAGG = 1. - self.DET_SAND - self.DET_SILT - self.DET_CLAY - self.DET_SMAGG
415  # Error check, may happen for soils with more sand. The fraction should not added up to 1.0
416  if self.DET_LGAGG < 0.:
417  self.DET_SAND /= (1. - self.DET_LGAGG)
418  self.DET_SILT /= (1. - self.DET_LGAGG)
419  self.DET_CLAY /= (1. - self.DET_LGAGG)
420  self.DET_SMAGG /= (1. - self.DET_LGAGG)
421  self.DET_LGAGG = 0.
422  # initialize water/drainage coefs for each soil layer
423  sumpor = 0.
424  self.SOL_SUMUL = 0.
425  self.SOL_SUMAWC = 0.
426  self.SOL_SUMWP = 0.
427  for i in range(self.SOILLAYERS):
428  pormm = self.POROSITY[i] * self.SOILTHICK[i]
429  sumpor += pormm
430  self.SOL_UL.append((self.POROSITY[i] - self.WILTINGPOINT[i]) * self.SOILTHICK[i])
431  self.SOL_SUMUL += self.SOL_UL[i]
432  self.SOL_AWC.append((self.FIELDCAP[i] - self.WILTINGPOINT[i]) * self.SOILTHICK[i])
433  self.SOL_SUMAWC += self.SOL_AWC[i]
434  self.SOL_HK.append((self.SOL_UL[i] - self.SOL_AWC[i]) / self.CONDUCTIVITY[i])

510  @staticmethod
511  def getsoiltexture_usda(clay, silt, sand):
512  """
513  The soil texture code system is from WetSpa Extension and SWAT model which is
514  based on the soil texture triangle developed by USDA.
515  The unit is percentage, silt + sand + clay [+ Rock] = 100.
516  The corresponding default soil parameters (e.g. Ks, porosity) are stored in
517  `seims/database/SoilLookup.txt`.
518  :param clay: clay content percentage
519  :param silt: silt content percentage
520  :param sand: sand content percentage
521  :return: [Soil texture ID, Hydrological soil group, USLE K factor]
522  """
523  if clay >= 40 >= silt and sand <= 45:
524  return [12, 1, 0.22] # clay / nian tu
525  elif clay >= 40 and silt >= 40:
526  return [11, 1, 0.26] # silt caly / fen nian tu
527  elif clay >= 35 and sand >= 45:
528  return [10, 1, 0.28] # sandy clay / sha nian tu
529  elif clay >= 25 and 20 <= sand <= 45:
530  return [9, 2, 0.3] # clay loam / nian rang tu
531  elif clay >= 25 and sand <= 20:
532  return [8, 2, 0.32] # silt clay loam / fen zhi nian rang tu
533  elif clay >= 20 and silt <= 30 and sand >= 45:
534  return [7, 2, 0.2] # sandy clay loam / sha zhi nian rang tu
535  elif clay >= 10 and 30 <= silt <= 50 and sand <= 50:
536  return [6, 3, 0.3] # loam / rang tu
537  elif 50 <= silt <= 80 or clay >= 15 and silt >= 80:
538  return [4, 3, 0.38] # silt loam / fen zhi rang tu
539  elif silt >= 80 and clay <= 15:
540  return [5, 3, 0.34] # silt / fen tu
541  elif clay <= 10 and sand <= 50 or 50 <= sand <= 80:
542  return [3, 4, 0.13] # sandy loam / sha zhi rang tu
543  elif sand <= 90:
544  return [2, 4, 0.04] # loamy sand / rang zhi sha tu
545  else:
546  return [1, 4, 0.02] # sand / sha tu

664  """Initialize soil moisture fraction of field capacity, based on TWI"""
665  acc_R = RasterUtilClass.ReadRaster(acc_file)
666  dataAcc = acc_R.data
667  xsize = acc_R.nCols
668  ysize = acc_R.nRows
669  noDataValue = acc_R.noDataValue

671  geotrans = acc_R.geotrans
672  dx = acc_R.dx
673  dataSlope = RasterUtilClass.ReadRaster(slope_file).data
674  cellArea = dx * dx
675 
676  # TWI, ln(acc_file/tan(slp))
677  def wiGridCal(accvalue, slpvalue):
678  if abs(accvalue - noDataValue) < UTIL_ZERO:
679  return DEFAULT_NODATA
680  else:
681  if abs(slpvalue) < MINI_SLOPE:
682  slpvalue = MINI_SLOPE
683  return math.log((accvalue + 1.) * cellArea / slpvalue)
684 
685  wiGridCal_numpy = numpy.frompyfunc(wiGridCal, 2, 1)
686  wiGrid = wiGridCal_numpy(dataAcc, dataSlope)
687  # wiGrid_valid = numpy.where(acc_R.validZone, wiGrid, numpy.nan)
688  # wiMax = numpy.nanmax(wiGrid_valid)
689  # wiMin = numpy.nanmin(wiGrid_valid)
690  # WARNING: numpy.nanmax and numpy.nanmin are un-stable in Linux, so
691  # replaced by the for loops. By LJ
692  wiMax = -numpy.inf
693  wiMin = numpy.inf
694  for i in range(0, ysize):
695  for j in range(0, xsize):
696  if wiMax < wiGrid[i][j]:
697  wiMax = wiGrid[i][j]
698  if DEFAULT_NODATA != wiGrid[i][j] < wiMin:
699  wiMin = wiGrid[i][j]
700  # print "TWIMax:%f, TWIMin:%f" % (wiMax, wiMin)
701  soilMoisFrMin = 0.6 # minimum relative saturation
702  soilMoisFrMax = 1.0
703 
704  wiUplimit = wiMax
705  a = (soilMoisFrMax - soilMoisFrMin) / (wiUplimit - wiMin)
706  b = soilMoisFrMin - a * wiMin
707 
708  def moistureCal(acc, wigrid):
709  if abs(acc - noDataValue) < UTIL_ZERO:
710  return DEFAULT_NODATA
711  else:
712  tmp = a * wigrid + b
713  if tmp > soilMoisFrMax:
714  return soilMoisFrMax
715  elif tmp < soilMoisFrMin:
716  return soilMoisFrMin

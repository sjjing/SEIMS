124 int GWaterReservoir::Execute(void) {
125  InitOutputs();
126  CheckInputData();
127 #pragma omp parallel for
128  for (int i = 0; i <= m_nReaches; i++) {
129  m_percSubbasin[i] = 0.f;
130  }
131 
132  // get percolation for each subbasin
133  for (int i = 0; i < m_nCells; i++) {
134 #ifdef MULTIPLY_REACHES
135  int subbasinId = (int) m_subbasin[i];
136 #else
137  int subbasinId = 1;
138 #endif
139  m_percSubbasin[subbasinId] += m_recharge[i];
140  }
141 
142  //float sum = 0.f;
143 #pragma omp parallel for //reduction(+:sum)
144  for (int i = 1; i <= m_nReaches; i++) {
145  float percolation = m_percSubbasin[i] * (1 - m_deepCoefficient) / m_nCellsSubbasin[i];
146  // depth of groundwater runoff(mm)
147  float outFlowDepth = m_recessionCoefficient * pow(m_storage[i], m_recessionExponent);
148  // groundwater flow out of the subbasin at time t (m3/s)
149  m_qg[i] = outFlowDepth / 1000 * m_nCellsSubbasin[i] * m_CellWidth * m_CellWidth / m_dt;
150  //sum = sum + m_qg[i];
151 
152  // water balance (mm)
153  m_storage[i] += percolation - outFlowDepth;
154  }
155  //m_qg[0] = sum;
156 
157  return 0;
158 }

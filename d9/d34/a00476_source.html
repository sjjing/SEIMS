59 int PETHargreaves::Execute() {
60  if (!this->CheckInputData()) return false;
61  initialOutputs();
62  m_jday = JulianDay(this->m_date);
63  //cout<<m_jday<<","m_tMean[0]<<","<<m_tMax[0]<<","<<m_tMin[0]<<endl;
64 #pragma omp parallel for
65  for (int i = 0; i < m_nCells; ++i) {
67  /* update base zero total heat units, src code from SWAT, subbasin.f
68  if (tmpav(j) > 0. .and. phutot(hru_sub(j)) > 0.01) then
69  phubase(j) = phubase(j) + tmpav(j) / phutot(hru_sub(j))
70  end if*/
71  if (m_jday == 1) {
72  m_phuBase[i] = 0.f;
73  }
74  if (m_tMean[i] > 0.f && m_phutot[i] > 0.01f) {
75  m_phuBase[i] += m_tMean[i] / m_phutot[i];
76  }
77  MaxSolarRadiation(m_jday, m_cellLat[i], m_dayLen[i], m_srMax);
79  float latentHeat = 2.501f - 0.002361f * m_tMean[i];
82  float h0 = m_srMax * 37.59f / 30.0f;
85  float petValue = m_HCoef_pet * h0 * pow(abs(m_tMax[i] - m_tMin[i]), 0.5f)
86  * (m_tMean[i] + 17.8f) / latentHeat;
87  m_pet[i] = m_petFactor * max(0.0f, petValue);
89  float satVaporPressure = SaturationVaporPressure(m_tMean[i]);
90  float actualVaporPressure = 0.f;
91  if (m_rhd[i] > 1) {
92  actualVaporPressure = m_rhd[i] * satVaporPressure * 0.01f;
93  } else {
94  actualVaporPressure = m_rhd[i] * satVaporPressure;
95  }
96  m_vpd[i] = satVaporPressure - actualVaporPressure;
97  }
98  return 0;
99 }

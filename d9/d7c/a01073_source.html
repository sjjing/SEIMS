137  m_chCellCount[subi] += 1;
138  }
139 
140  for (int i = 1; i <= m_nReaches; i++) {
141  m_chDaylen[i] /= m_chCellCount[i];
142  m_chSr[i] /= m_chCellCount[i];

496  for (int i = 0; i <= m_nReaches; i++) {

510  vector<int> m_ptSrcMgtSeqs = it->second->GetPointSrcMgtSeqs();

514  // 1. looking for management operations from m_pointSrcMgtMap
515  for (vector<int>::iterator seqIter = m_ptSrcMgtSeqs.begin(); seqIter != m_ptSrcMgtSeqs.end(); seqIter++) {
516  PointSourceMgtParams *curPtMgt = m_pointSrcMgtMap.at(*seqIter);
517  // 1.1 If current day is beyond the date range, then continue to next management
518  if (curPtMgt->GetStartDate() != 0 && curPtMgt->GetEndDate() != 0) {
519  if (m_date < curPtMgt->GetStartDate() || m_date > curPtMgt->GetEndDate()) {
520  continue;
521  }
522  }
523  // 1.2 Otherwise, get the nutrient concentration, mg/L
524  float per_wtr = curPtMgt->GetWaterVolume();
525  float per_no3 = curPtMgt->GetNO3();
526  float per_nh4 = curPtMgt->GetNH4();
527  float per_orgn = curPtMgt->GetOrgN();

531  // 1.3 Sum up all point sources

541  m_ptSolPToCh[curSubID] += per_solp * cvt;

544  m_ptTPToCh[curSubID] += (per_solp + per_orgP) * cvt;
545  }
546  }
547  }
548  }
549  // sum up point sources loadings of all subbasins
550  for (int i = 1; i <= m_nReaches; i++) {
551  m_ptTNToCh[0] += m_ptTNToCh[i];

352 double SubbasinIUHCalculator::IUHti(double delta0, double t00, double ti) {
353  double tmp1 = 1 / (delta0 * sqrt(2 * 3.1416 * pow(ti, 3.0) / pow(t00, 3.0)));
354  double tmp2 = pow(ti - t00, 2.0) / (2.0 * pow(delta0, 2.0) * ti / t00);
355  return tmp1 * exp(-tmp2);
356 }

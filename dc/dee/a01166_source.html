94 int clsPI_SVSC::Execute() {
95  //check input data
96  CheckInputData();
98  initialOutputs();
99 
100  int julian = JulianDay(m_date);
101 #pragma omp parallel for
102  for (int i = 0; i < this->m_nCells; i++) {
103  if (m_P[i] > 0.f) {
104 #ifdef STORM_MODE
105  m_P[i] = m_P[i] * m_hilldt / 3600.f * cos(atan(m_slope[i]));
107 #endif // STORM_MODE
108  //interception storage capacity
109  float degree = 2.f * PI * (julian - 87.f) / 365.f;
111  float min = m_minSt[i];
112  float max = m_maxSt[i];
113  float capacity = min + (max - min) * pow(0.5f + 0.5f * sin(degree), m_Pi_b);
114 
115  //interception, currently, m_st[i] is storage of (t-1) time step
116  float availableSpace = capacity - m_st[i];
117  if (availableSpace < 0) {
118  availableSpace = 0.f;
119  }
120 
121  if (availableSpace < m_P[i]) {
122  m_interceptionLoss[i] = availableSpace;
123  } else {
124  m_interceptionLoss[i] = m_P[i];
125  }
126 
127  //net precipitation
128  m_netPrecipitation[i] = m_P[i] - m_interceptionLoss[i];
129  m_st[i] += m_interceptionLoss[i];
130  } else {
131  m_interceptionLoss[i] = 0.f;
132  m_netPrecipitation[i] = 0.f;
133  }
134 #ifndef STORM_MODE
135  //evaporation
136  if (m_st[i] > m_PET[i]) {
137  m_evaporationLoss[i] = m_PET[i];
138  } else {
139  m_evaporationLoss[i] = m_st[i];
140  }
141  m_st[i] -= m_evaporationLoss[i];
142 #endif
143  }
144  return 0;
145 }

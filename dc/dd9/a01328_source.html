176 void DiffusiveWave::ChannelFlow(int iReach, int iCell, int id) {
177 
178  float qUp = 0.f;
179  float hUp = 0.f;
180 
181  if (iReach == 0 && iCell == 0) {
182  qUp = m_qUpReach;
183  }
184 
185  // inflow from upstream channel
186  if (iCell == 0)// inflow of this cell is the last cell of the upstream reach
187  {
188  for (size_t i = 0; i < m_reachUpStream[iReach].size(); ++i) {
189  int upReachId = m_reachUpStream[iReach][i];
190  if (upReachId >= 0) {
191  int upCellsNum = m_reachs[upReachId].size();
192  int upCellId = m_reachs[upReachId][upCellsNum - 1];
193  qUp += m_qCh[upReachId][upCellsNum - 1];
194 
195  float hWater = m_elevation[upCellId] + m_hCh[upReachId][upCellsNum - 1];
196  if (hWater > hUp) {
197  hUp = hWater;
198  }
199  }
200  }
201  } else {
202  qUp = m_qCh[iReach][iCell - 1];
203  hUp = m_hCh[iReach][iCell - 1];
204  }
205 
206  //float b = 0.6f;
207  float h = m_hCh[iReach][iCell];
208  float dx = m_flowLen[iReach][iCell];
209 
210  float qLat = m_prec[id] / 1000.f * m_chWidth[id];
211  if (m_qs != NULL) {
212  qLat += m_qs[id] / dx;
213  }
214  if (m_qi != NULL) {
215  qLat += m_qi[id] / dx;
216  }
217 
218  if (qLat < MIN_FLUX && qUp < MIN_FLUX) {
219  m_hCh[iReach][iCell] = 0.f;
220  m_qCh[iReach][iCell] = 0.f;
221  return;
222  }
223 
224  float perim = 2.f * h + m_chWidth[iReach];
225  float sf = (hUp - m_elevation[id] - h) / dx;
226  if (sf < MINI_SLOPE) {
227  sf = MINI_SLOPE;
228  }
229  float c = 1.f / 3600.f * m_manningScalingFactor * m_reachN[iReach] * pow(perim, _23) / sqrt(sf);
230  c = pow(c, 0.6f);
231 
232  float d = 1.f;
233  int counter = 0;
234  float qLast = m_qCh[iReach][iCell];
235  float qNew = qLast;
236  if (qNew < MIN_FLUX) {
237  qNew = qLat;
238  }
239 
240  while (abs(d) > MIN_FLUX && counter < 10) {
241  d = (qNew * m_dt / dx + c * Power(qNew, 0.6f) - qUp * m_dt / dx - c * Power(qLast, 0.6f) - qLat * m_dt) /
242  (m_dt / dx + c * 0.6f / Power(qNew, 0.4f));
243 
244  //if(d != d)
245  // int test = 1;
246  qNew -= d;
247  counter++;
248  }
249 
250  if (qNew < 0.f) {
251  qNew = 0.f;
252  }
253 
254  float qAvail = m_hCh[iReach][iCell] * m_chWidth[iReach] * dx / m_dt + qLat * dx + qUp;
255 
256  if (qNew > qAvail) {
257  m_qCh[iReach][iCell] = qAvail;
258  m_hCh[iReach][iCell] = 0.f;
259  } else {
260  m_qCh[iReach][iCell] = qNew;//m_hCh[iReach][iCell]
261  m_hCh[iReach][iCell] = c * pow(qNew, 0.6f) / m_chWidth[iReach];
262  //float hh = (qUp + qLat*dx - qNew)*m_dt/(m_chWidth[iReach]*dx) + m_hCh[iReach][iCell];
263  }

267 int DiffusiveWave::Execute() {
268  //check the data
269  CheckInputData();
270 
271  initialOutputs();
272  //Output1DArray(m_nCells, m_prec, "f:\\p2.txt");
273  map < int, vector < int > > ::iterator
274  it;
275  for (it = m_reachLayers.begin(); it != m_reachLayers.end(); it++) {
276  // There are not any flow relationship within each routing layer.
277  // So parallelization can be done here.
278  int nReaches = it->second.size();
279  // the size of m_reachLayers (map) is equal to the maximum stream order
280 #pragma omp parallel for
281  for (int i = 0; i < nReaches; ++i) {
282  int reachIndex = it->second[i]; // index in the array
283  vector<int> &vecCells = m_reachs[reachIndex];
284  int n = vecCells.size();
285  for (int iCell = 0; iCell < n; ++iCell) {
286  ChannelFlow(reachIndex, iCell, vecCells[iCell]);
287  }
288  m_qSubbasin[reachIndex] = m_qCh[reachIndex][n - 1];
289  }
290  }

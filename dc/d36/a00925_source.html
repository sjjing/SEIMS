27

37  cxpoint2 = random.randint(1, size)
38  if cxpoint2 < cxpoint1: # Swap the two cx points
39  cxpoint1, cxpoint2 = cxpoint2, cxpoint1

42  cs2 = cxpoint2 / sp_num

51  break # avoid change the entire genes

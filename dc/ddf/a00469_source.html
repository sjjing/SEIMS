141 void IKW_REACH::initialOutputs() {
142  if (m_nreach <= 0) {
143  throw ModelException("IKW_REACH", "initialOutputs", "The cell number of the input can not be less than zero.");
144  }
145 
146  if (m_reachLayers.empty()) {
147  CheckInputData();
148  for (int i = 1; i <= m_nreach; i++) {
149  int order = (int) m_chOrder[i];
150  m_reachLayers[order].push_back(i);
151  }
152  }
153 
154  //initial channel storage
155  if (m_chStorage == NULL) {
156  m_chStorage = new float[m_nreach + 1];
157  m_qIn = new float[m_nreach + 1];
158  m_qOut = new float[m_nreach + 1];
159  m_bankStorage = new float[m_nreach + 1];
160  m_seepage = new float[m_nreach + 1];
161  m_qsCh = new float[m_nreach + 1];
162  m_qiCh = new float[m_nreach + 1];
163  m_qgCh = new float[m_nreach + 1];
164  m_chWTdepth = new float[m_nreach + 1];
165 
166 #pragma omp parallel for
167  for (int i = 1; i <= m_nreach; i++) {
168  float qiSub = 0.f;
169  float qgSub = 0.f;
170  if (m_qiSub != NULL) {

190 int IKW_REACH::Execute() {
191  initialOutputs();
192 
193  map < int, vector < int > > ::iterator
194  it;
195  for (it = m_reachLayers.begin(); it != m_reachLayers.end(); it++) {
196  // There are not any flow relationship within each routing layer.
197  // So parallelization can be done here.
198  int nReaches = it->second.size();
199  // the size of m_reachLayers (map) is equal to the maximum stream order
200 #pragma omp parallel for
201  for (int i = 0; i < nReaches; ++i) {
202  int reachIndex = it->second[i]; // index in the array
203  ChannelFlow(reachIndex);
204  }
205  }
206 
207  return 0;

444 float IKW_REACH::GetNewQ(float qIn, float qLast, float surplus, float alpha, float dt, float dx) {
445  /* Using Newton-Raphson Method */
446  float ab_pQ, dtX, C; //auxillary vars
447  int count;
448  float Qkx; //iterated discharge, becomes Qnew
449  float fQkx; //function
450  float dfQkx; //derivative
451  const float _epsilon = 1e-12f;
452  const float beta = 0.6f;
453 
454  /* if no input then output = 0 */
455  if ((qIn + qLast) <= -surplus * dx)//0)
456  {
457  //itercount = -1;
458  return (0);
459  }
460 
461  /* common terms */
462  ab_pQ = alpha * beta * pow(((qLast + qIn) / 2), beta - 1);
463  // derivative of diagonal average (space-time)
464 
465  dtX = dt / dx;
466  C = dtX * qIn + alpha * pow(qLast, beta) + dt * surplus;
467  //dt/dx*Q = m3/s*s/m=m2; a*Q^b = A = m2; surplus*dt = s*m2/s = m2
468  //C is unit volume of water
469  // first gues Qkx
470  Qkx = (dtX * qIn + qLast * ab_pQ + dt * surplus) / (dtX + ab_pQ);
471 
472  // VJ 050704, 060830 infil so big all flux is gone
473  //VJ 110114 without this de iteration cannot be solved for very small values
474  if (Qkx < MIN_FLUX) {
475  //itercount = -2;
476  return (0);
477  }
478 
479  Qkx = max(Qkx, MIN_FLUX);
480 
481  count = 0;
482  do {
483  fQkx = dtX * Qkx + alpha * pow(Qkx, beta) - C; /* Current k */
484  dfQkx = dtX + alpha * beta * pow(Qkx, beta - 1); /* Current k */
485  Qkx -= fQkx / dfQkx; /* Next k */
486  Qkx = max(Qkx, MIN_FLUX);
487  count++;
488  //qDebug() << count << fQkx << Qkx;
489  } while (fabs(fQkx) > _epsilon && count < MAX_ITERS_CH);
490 
491  if (Qkx != Qkx) {
492  throw ModelException("IKW_OL", "GetNewQ", "Error in iteration!");
493  }
494 
495  //itercount = count;
496  return Qkx;
497 }

499 void IKW_REACH::ChannelFlow(int i) {
500  float st0 = m_chStorage[i];
501 
502  float qiSub = 0.f;
503  if (m_qiSub != NULL) {
504  qiSub = m_qiSub[i];
505  }
506  float qgSub = 0.f;
507  if (m_qgSub != NULL) {
508  qgSub = m_qgSub[i];
509  }
510 
512  // first add all the inflow water
513  // 1. water from this subbasin
514  float qIn = m_qsSub[i] + qiSub + qgSub;
515 
516  // 2. water from upstream reaches
517  float qsUp = 0.f;
518  float qiUp = 0.f;
519  float qgUp = 0.f;
520  for (size_t j = 0; j < m_reachUpStream[i].size(); ++j) {
521  int upReachId = m_reachUpStream[i][j];
522  qsUp += m_qsCh[upReachId];
523  qiUp += m_qiCh[upReachId];
524  qgUp += m_qgCh[upReachId];
525  }
526  qIn += qsUp + qiUp + qgUp;
527  qIn +=
528  m_qUpReach; // m_qUpReach is zero for not-parallel program and qsUp, qiUp and qgUp are zero for parallel computing
529 
530  // 3. water from bank storage
531  float bankOut = m_bankStorage[i] * (1 - exp(-m_aBank));
532 
533  m_bankStorage[i] -= bankOut;
534  qIn += bankOut / m_dt;
535 
536  // add inflow water to storage
537  m_chStorage[i] += qIn * m_dt;
538 
540  // then subtract all the outflow water
541  // 1. transmission losses to deep aquifer, which is lost from the system
542  // the unit of kchb is mm/hr
543  float seepage = m_Kchb / 1000.f / 3600.f * m_chWidth[i] * m_chLen[i] * m_dt;
544  if (qgSub < 0.001f) {
545  if (m_chStorage[i] > seepage) {
546  m_seepage[i] = seepage;
547  m_chStorage[i] -= seepage;
548  } else {
549  m_seepage[i] = m_chStorage[i];
550  m_chStorage[i] = 0.f;
551  m_qOut[i] = 0.f;
552  m_qsCh[i] = 0.f;
553  m_qiCh[i] = 0.f;
554  m_qgCh[i] = 0.f;
555  return;
556  }
557  } else {
558  m_seepage[i] = 0.f;
559  }
560 
561  // 2. calculate transmission losses to bank storage
562  float dch = m_chStorage[i] / (m_chWidth[i] * m_chLen[i]);
563  float bankInLoss = 2 * m_Kbank / 1000.f / 3600.f * dch * m_chLen[i] * m_dt; // m3/s
564  bankInLoss = 0.f;
565  if (m_chStorage[i] > bankInLoss) {
566  m_chStorage[i] -= bankInLoss;
567  } else {
568  bankInLoss = m_chStorage[i];
569  m_chStorage[i] = 0.f;
570  }
571  // water balance of the bank storage
572  // loss the water from bank storage to the adjacent unsaturated zone and groundwater storage
573  float bankOutGw = m_bankStorage[i] * (1 - exp(-m_bBank));
574  bankOutGw = 0.f;
575  m_bankStorage[i] = m_bankStorage[i] + bankInLoss - bankOutGw;
576  if (m_gwStorage != NULL) {

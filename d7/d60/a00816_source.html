148 int SUR_CN::Execute() {
149  CheckInputData();
150  initialOutputs();
151 
152  float cnday;
153  float pNet, surfq, infil;
154 
155 #pragma omp parallel for
156  for (int iCell = 0; iCell < m_nCells; iCell++) {
157  //initialize the variables
158  surfq = 0.0f;
159  infil = 0.0f;
160  pNet = 0.0f;
161  float pcp = m_P_NET[iCell]; //rainfall - interception
162  float dep = 0.0f; //depression storage
163  float snm = 0.0f; //snowmelt
164  float t = 0.0f; // air temperature
165 
166  //set values to variables
167  if (m_SD == NULL) // the depression storage module is not available, set the initial depression storage
168  {
169  //dep = m_Depre_in * m_Depression[iCell];
170  dep = 0.f;
171  } else {
172  dep = m_SD[iCell];
173  }
174 
175  if (m_SM == NULL) {
176  snm = 0.f;
177  } else {
178  snm = m_SM[iCell];
179  }
180 
182  t = m_tMean[iCell];
183  //account for the effects of snowmelt and soil temperature
184  // snow, without snow melt
185  if (t <= m_Tsnow) {
186  pNet = 0.0f;
187  //m_SA[iCell] += pcp;
188  }
189  // rain on snow, no snow melt
190  else if (t > m_Tsnow && t <= m_T0 && m_SA != NULL && m_SA[iCell] > pcp) {
191  pNet = 0.0f;
192  //m_SA[iCell] += pcp;
193  } else {
194  pNet = pcp + dep + snm; // default
195  }
196 
197  if (pNet > 0.0f) {
198  float sm = 0.f;
199  float por = 0.f;
200  //float aboveDepth = 0.f;
201 
202  int curSoilLayers = -1, j;
203  m_upSoilDepth[0] = m_soilDepth[iCell][0];
204  for (j = 1; j < m_nSoilLayers; j++) {
205  if (!FloatEqual(m_soilDepth[iCell][j], NODATA_VALUE)) {
206  m_upSoilDepth[j] = m_soilDepth[iCell][j] - m_soilDepth[iCell][j - 1];
207  } else {
208  break;
209  }
210  }
211  curSoilLayers = j;
212 
213  for (j = 0; j < curSoilLayers; j++) {
214  sm += m_soilMoisture[iCell][j] * m_upSoilDepth[j];
215  por += m_porosity[iCell][j] * m_upSoilDepth[j];
216  //aboveDepth += m_depth[j];
217  }
218  //sm += m_soilMoisture[i][m_nSoilLayers - 1] * (m_rootDepth[i] - aboveDepth);
219  //por += m_porosity[i][m_nSoilLayers - 1] * (m_rootDepth[i] - aboveDepth);
220 
221  sm /= por;
222  sm = min(sm, 1.0f);
223 
224  //for frozen soil
225  if (m_TS[iCell] <= m_Tsoil && sm >= m_Sfrozen * por) {
226  m_PE[iCell] = pcp + snm;
227  m_INFIL[iCell] = 0.0f;
228  }
229  //for saturation overland flow
230  else if (sm > por) {
231  m_PE[iCell] = pcp + snm;
232  m_INFIL[iCell] = 0.0f;
233  }
234  //for CN method
235  else {
236  cnday = Calculate_CN(sm, iCell);
237  float bb, pb;
238  float s = 0.0f;
239  bb = 0.0f;
240  pb = 0.0f;
241  s = 25400.0f / cnday - 254.0f;
242  bb = 0.2f * s;
243  pb = pNet - bb;
244  // calculate infiltration and excess rainfall
245  if (pb > 0.0f) {
246  surfq = pb * pb / (pNet + 0.8f * s);
247  }
248 
249  if (surfq < 0.0f) {
250  surfq = 0.0f;
251  }
252 
253  infil = pNet - surfq;
254 
255  // set output value
256  m_INFIL[iCell] = infil;
257  m_PE[iCell] = pcp + snm - infil; //excess precipitation could be negative
258  }
259 
260  if (m_INFIL[iCell] < 0) {
261  m_INFIL[iCell] = 0.f;
262  m_PE[iCell] = pcp + snm - m_INFIL[iCell];
263  }
264 
265  // check the output data
266  if (m_INFIL[iCell] != m_INFIL[iCell] || m_INFIL[iCell] < 0.0f) {
267  cout << m_INFIL[iCell] << endl;
268  throw ModelException(MID_SUR_CN,
269  "Execute",
270  "Output data error: infiltration is less than zero. :\n Please contact the module developer. ");
271  }
272 
273  } else {
274  m_PE[iCell] = 0.0f;
275  m_INFIL[iCell] = 0.0f;
276  }
277  }
278  return 0;

370 float SUR_CN::Calculate_CN(float sm, int cell) {
371  float sw, s, CNday, xx;
372 
373  s = 0.;
374  sw = sm * m_rootDepth[cell];
375  xx = m_w1[cell] - m_w2[cell] * sw;
376  if (xx < -20.f) {
377  xx = -20.f;
378  }
379  if (xx > 20.f) {
380  xx = 20.f;
381  }
382  // traditional CN method (function of soil water)
383  if ((sw + exp(xx)) > 0.001f) {
384  s = m_sMax[cell] * (1.f - sw / (sw + exp(xx))); //2:1.1.6
385  }
386  CNday = 25400.f / (s + 254.f); //2:1.1.11
387  return CNday;
388 }

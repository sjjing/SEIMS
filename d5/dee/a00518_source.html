268 #pragma omp parallel for

272  }
274  if (m_conv_wt == NULL) {
275  Initialize2DArray(m_nCells, m_soilLayers, m_conv_wt, 0.f);
276 #pragma omp parallel for
277  for (int i = 0; i < m_nCells; i++) {
278  for (int k = 0; k < (int) m_nSoilLayers[i]; k++) {
279  float wt1 = 0.f;
280  float conv_wt = 0.f;
281  // mg/kg => kg/ha
282  wt1 = m_sol_bd[i][k] * m_sol_thick[i][k] / 100.f;
283  // kg/kg => kg/ha
284  conv_wt = 1.e6f * wt1;
285  m_conv_wt[i][k] = conv_wt;
286  }
287  }
288  }
289  // initial input soil chemical in first run
290  if (m_sol_no3 == NULL) Initialize2DArray(m_nCells, m_soilLayers, m_sol_no3, 0.f);
291  if (m_sol_fon == NULL || m_sol_fop == NULL || m_sol_aorgn == NULL ||
292  m_sol_actp == NULL || m_sol_stap == NULL) {
293  Initialize2DArray(m_nCells, m_soilLayers, m_sol_fon, 0.f);
294  Initialize2DArray(m_nCells, m_soilLayers, m_sol_fop, 0.f);
295  Initialize2DArray(m_nCells, m_soilLayers, m_sol_aorgn, 0.f);
296  Initialize2DArray(m_nCells, m_soilLayers, m_sol_actp, 0.f);
297  Initialize2DArray(m_nCells, m_soilLayers, m_sol_stap, 0.f);
298 
299 #pragma omp parallel for
300  for (int i = 0; i < m_nCells; i++) {
301  // fresh organic P / N
302  m_sol_fop[i][0] = m_sol_cov[i] * .0010f;
303  m_sol_fon[i][0] = m_sol_cov[i] * .0055f;
304 
305  for (int k = 0; k < (int) m_nSoilLayers[i]; k++) {
306  //float wt1 = 0.f;
307  //float conv_wt = 0.f;
309  //wt1 = m_sol_bd[i][k] * m_sol_thick[i][k] / 100.f;
311  //conv_wt = 1.e6f * wt1;
312  float wt1 = m_conv_wt[i][k] * 1.e-6f;
314  if (m_sol_no3[i][k] <= 0.f) {
315  m_sol_no3[i][k] = 0.f;
316  float zdst = 0.f;
317  zdst = exp(-m_sol_z[i][k] / 1000.f);
318  m_sol_no3[i][k] = 10.f * zdst * 0.7f;
319  m_sol_no3[i][k] *= wt1; // mg/kg => kg/ha
320  //if(k == 0) outfile << m_sol_no3[i][k];
321  }
323  if (m_sol_orgn[i][k] <= 0.f) {
324  // CN ratio changed back to 14
325  m_sol_orgn[i][k] = 10000.f * (m_sol_cbn[i][k] / 14.f) * wt1;
326  }
327  // assume C:N ratio of 10:1
328  // nitrogen active pool fraction (nactfr)
329  float nactfr = .02f;
330  m_sol_aorgn[i][k] = m_sol_orgn[i][k] * nactfr;
331  m_sol_orgn[i][k] *= (1.f - nactfr);
332 
333  // currently not used
334  //sumorgn = sumorgn + m_sol_aorgn[i][k] + m_sol_orgn[i][k] + m_sol_fon[i][k];
335 
336  if (m_sol_orgp[i][k] <= 0.f) {
337  // assume N:P ratio of 8:1
338  m_sol_orgp[i][k] = 0.125f * m_sol_orgn[i][k];
339  }
340 
341  if (m_sol_solp[i][k] <= 0.f) {
342  // assume initial concentration of 5 mg/kg
343  m_sol_solp[i][k] = 5.f * wt1;
344  }
345 
346  float solp = 0.f;
347  float actp = 0.f;
348  float psp = m_psp;
349  if (m_solP_model == 0)// Set active pool based on dynamic PSP MJW
350  {
351  // Allow Dynamic PSP Ratio
352  if (m_conv_wt[i][k] != 0) solp = (m_sol_solp[i][k] / m_conv_wt[i][k]) * 1000000.f;
353  } else {
354  throw ModelException(MID_NUTR_TF, "initialOutputs",
355  "Please check the bulk density and soil thickness data.");
356  if (m_sol_clay[i][k] > 0.f) {
357  psp = -0.045f * log(m_sol_clay[i][k]) + (0.001f * solp);
358  psp = psp - (0.035f * m_sol_cbn[i][k]) + 0.43f;
359  } else {
360  psp = 0.4f;
361  }
362  // Limit PSP range
363  if (psp < .05f) { psp = 0.05f; }
364  else if (psp > 0.9f) psp = 0.9f;
365  }
366 
367  m_sol_actp[i][k] = m_sol_solp[i][k] * (1.f - psp) / psp;
368 
369  if (m_solP_model == 0) // Set Stable pool based on dynamic coefficient
370  { // From White et al 2009
371  // convert to concentration for ssp calculation
372  actp = m_sol_actp[i][k] / m_conv_wt[i][k] * 1000000.f;
373  solp = m_sol_solp[i][k] / m_conv_wt[i][k] * 1000000.f;
374  // estimate Total Mineral P in this soil based on data from sharpley 2004
375  float ssp = 0.;
376  ssp = 25.044f * pow((actp + solp), -0.3833f);
377  // limit SSP Range
378  if (ssp > 7.f) ssp = 7.f;
379  if (ssp < 1.f) ssp = 1.f;
380  // define stableP
381  m_sol_stap[i][k] = ssp * (m_sol_actp[i][k] + m_sol_solp[i][k]);
382  } else {
383  // The original code
384  m_sol_stap[i][k] = 4.f * m_sol_actp[i][k];

540 void Nutrient_Transformation::Mineralization_StaticCarbonMethod(int i) {
541  //soil layer (k)
542  for (int k = 0; k < (int) m_nSoilLayers[i]; k++) {
543  //soil layer used to compute soil water and soil temperature factors
544  int kk = 0;
545  if (k == 0) {
546  kk = 1;
547  } else {
548  kk = k;
549  }
550  float sut = 0.f;
551  // mineralization can occur only if temp above 0 deg
552  if (m_sote[i] > 0) {
553  // compute soil water factor (sut)
554  sut = 0.f;
555  if (m_soilStorage[i][kk] < 0) {
556  m_soilStorage[i][kk] = 0.0000001f;
557  }
558  sut = 0.1f + 0.9f * sqrt(m_soilStorage[i][kk] / m_sol_awc[i][kk]);
559  sut = max(0.05f, sut);
560 
561  //compute soil temperature factor
562  //variable to hold intermediate calculation result (xx)
563  float xx = 0.f;
564  //soil temperature factor (cdg)
565  float cdg = 0.f;
566  xx = m_sote[i];
567  cdg = 0.9f * xx / (xx + exp(9.93f - 0.312f * xx)) + 0.1f;
568  cdg = max(0.1f, cdg);
569 
570  // compute combined factor
571  xx = 0.f;
572  //combined temperature/soil water factor (csf)
573  float csf = 0.f;
574  xx = cdg * sut;
575  if (xx < 0) {
576  xx = 0.f;
577  }
578  if (xx > 1000000) {
579  xx = 1000000.f;
580  }
581  csf = sqrt(xx);
582 
583  // compute flow from active to stable pools
584  //amount of nitrogen moving from active organic to stable organic pool in layer (rwn)
585  float rwn = 0.f;
586  rwn = 0.1e-4f * (m_sol_aorgn[i][k] * (1.f / m_nactfr - 1.f) - m_sol_orgn[i][k]);
587  if (rwn > 0.f) {
588  rwn = min(rwn, m_sol_aorgn[i][k]);
589  } else {
590  rwn = -(min(abs(rwn), m_sol_orgn[i][k]));
591  }
592  m_sol_orgn[i][k] = max(1.e-6f, m_sol_orgn[i][k] + rwn);
593  m_sol_aorgn[i][k] = max(1.e-6f, m_sol_aorgn[i][k] - rwn);
594 
595  // compute humus mineralization on active organic n
596  //amount of nitrogen moving from active organic nitrogen pool to nitrate pool in layer (hmn)
597  float hmn = 0.f;
598  hmn = m_cmn * csf * m_sol_aorgn[i][k];
599  hmn = min(hmn, m_sol_aorgn[i][k]);
600  // compute humus mineralization on active organic p
601  xx = 0.f;
602  //amount of phosphorus moving from the organic pool to the labile pool in layer (hmp)
603  float hmp = 0.f;
604  xx = m_sol_orgn[i][k] + m_sol_aorgn[i][k];

606  hmp = 1.4f * hmn * m_sol_orgp[i][k] / xx;
607  } else {

609  }
610  hmp = min(hmp, m_sol_orgp[i][k]);
611

613  m_sol_aorgn[i][k] = max(1.e-6f, m_sol_aorgn[i][k] - hmn);

615  m_sol_orgp[i][k] = m_sol_orgp[i][k] - hmp;
616  m_sol_solp[i][k] = m_sol_solp[i][k] + hmp;
617 
618  // compute residue decomp and mineralization of
619  // fresh organic n and p (upper two layers only)
620  //amount of nitrogen moving from fresh organic to nitrate(80%) and active organic(20%) pools in layer (rmn1)
621  float rmn1 = 0.f;
622  //amount of phosphorus moving from fresh organic to labile(80%) and organic(20%) pools in layer (rmp)
623  float rmp = 0.f;
624  if (k <= 2) {

626  float cnr = 0.f;
627  //the C:P ratio (cnr)
628  float cpr = 0.f;

630  float ca = 0.f;

654  float decr = 0.f;

730 void Nutrient_Transformation::Volatilization(int i) {
731  //soil layer (k)
732  float kk = 0.f;
733  for (int k = 0; k < (int) m_nSoilLayers[i]; k++) {
734  //nitrification/volatilization temperature factor (nvtf)
735  float nvtf = 0.f;
736  //Calculate nvtf, equation 3:1.3.1 in SWAT Theory 2009, p192
737  nvtf = 0.41f * (m_sote[i] - 5.f) / 10.f;
738  if (m_sol_nh4[i][k] > 0.f && nvtf >= 0.001f) {
739  float sw25 = 0.f;
740  float swwp = 0.f;
741  //nitrification soil water factor (swf)
742  float swf = 0.f;
743  //Calculate nvtf, equation 3:1.3.2 and 3:1.3.3 in SWAT Theory 2009, p192
744  sw25 = m_sol_wpmm[i][k] + 0.25f * m_sol_awc[i][k];
745  swwp = m_sol_wpmm[i][k] + m_soilStorage[i][k];
746  if (swwp < sw25) {
747  swf = (swwp - m_sol_wpmm[i][k]) / (sw25 - m_sol_wpmm[i][k]);
748  } else {
749  swf = 1.f;
750  }
751 
752  if (k == 0) {
753  kk = 0.f;
754  } else {
755  kk = m_sol_z[i][k - 1];
756  }
757  //depth from the soil surface to the middle of the layer (dmidl/mm)
758  float dmidl = 0.f;
759  //volatilization depth factor (dpf)
760  float dpf = 0.f;
761  //nitrification regulator (akn)
762  float akn = 0.f;
763  //volatilization regulator (akn)
764  float akv = 0.f;
765  //amount of ammonium converted via nitrification and volatilization in layer (rnv)
766  float rnv = 0.f;
767  //amount of nitrogen moving from the NH3 to the NO3 pool (nitrification) in the layer (rnit)
768  float rnit = 0.f;
769  //amount of nitrogen lost from the NH3 pool due to volatilization (rvol)
770  float rvol = 0.f;
771  //volatilization CEC factor (cecf)
772  float cecf = 0.15f;
773  //Calculate nvtf, equation 3:1.3.2 and 3:1.3.3 in SWAT Theory 2009, p192
774  dmidl = (m_sol_z[i][k] + kk) / 2.f;
775  dpf = 1.f - dmidl / (dmidl + exp(4.706f - 0.0305f * dmidl));
776  //Calculate rnv, equation 3:1.3.6, 3:1.3.7 and 3:1.3.8 in SWAT Theory 2009, p193
777  akn = nvtf * swf;
778  akv = nvtf * dpf * cecf;
779  rnv = m_sol_nh4[i][k] * (1.f - exp(-akn - akv));
780  //Calculate rnit, equation 3:1.3.9 in SWAT Theory 2009, p193
781  rnit = 1.f - exp(-akn);
782  //Calculate rvol, equation 3:1.3.10 in SWAT Theory 2009, p193
783  rvol = 1.f - exp(-akv);
784 
785  //calculate nitrification (NH3 => NO3)
786  if ((rvol + rnit) > 1.e-6f) {
787  //Calculate the amount of nitrogen removed from the ammonium pool by nitrification,
788  //equation 3:1.3.11 in SWAT Theory 2009, p193
789  rvol = rnv * rvol / (rvol + rnit);
790  //Calculate the amount of nitrogen removed from the ammonium pool by volatilization,
791  //equation 3:1.3.12 in SWAT Theory 2009, p194
792  rnit = rnv - rvol;
793  if (rnit < 0)rnit = 0.f;
794  m_sol_nh4[i][k] = max(1e-6f, m_sol_nh4[i][k] - rnit);
795  }
796  if (m_sol_nh4[i][k] < 0.f) {
797  rnit = rnit + m_sol_nh4[i][k];
798  m_sol_nh4[i][k] = 0.f;
799  }
800  //debug
801  //if (rnit != rnit)
802  // cout<<"cellid: "<<i<<", lyr: "<<k<<", akn: "<<akn<<", nvtf:"<<
803  // nvtf<<", swf: "<<swf<<", dpf: "<<dpf<<", cecf: "<<cecf<<endl;
804  m_sol_no3[i][k] = m_sol_no3[i][k] + rnit;
805  //calculate ammonia volatilization
806  m_sol_nh4[i][k] = max(1e-6f, m_sol_nh4[i][k] - rvol);
807  if (m_sol_nh4[i][k] < 0.f) {
808  rvol = rvol + m_sol_nh4[i][k];
809  m_sol_nh4[i][k] = 0.f;

77 int SSM_PE::Execute() {
78  this->CheckInputData();
79 
80  this->initialOutputs();
81  if (m_SR == NULL)
82  {
83  m_SR = new float[m_nCells];
84  for (int i = 0; i < this->m_nCells; i++) {
85  m_SR[i] = 0.f;
86  }
87  }
88  if (m_isInitial) {
89  int count = 0;
90  for (int i = 0; i < this->m_nCells; i++) {
91  if (this->m_tMean[i] < this->m_tsnow) {
92  this->m_SA[i] = this->m_swe0;
93  count++;
94  } //winter
95  else { this->m_SA[i] = 0.0f; } // other seasons
96  }
97 
98  m_swe = this->m_swe0 * count / this->m_nCells;
99  m_isInitial = false;
100  }
101 
102  if (this->m_lastSWE == NODATA_VALUE) this->m_lastSWE = this->m_swe;
103  if (m_swe < 0.01) //all cells have not snow, so snow sublimation is 0.
104  {
105  if (this->m_lastSWE >= 0.01) {
106  for (int rw = 0; rw < this->m_nCells; rw++) {
107  m_SE[rw] = 0.0f;
108  }
109  }
110  this->m_lastSWE = this->m_swe;
111  return true;
112  }
113 
114  for (int rw = 0; rw < this->m_nCells; rw++) {
115  float snow = this->m_SA[rw] + this->m_SR[rw];
116  float tmean = this->m_tMean[rw];
117  if (tmean < this->m_tsnow) snow += (1 + this->m_kblow) * this->m_Pnet[rw];
118  if (snow < 0.01) { this->m_SE[rw] = 0.0f; }
119  else {
120  if (tmean >= this->m_t0) {
121  this->m_SE[rw] = 0.0f; //if temperature is higher than t0, the sublimation is 0.
122  } else {
123  float se = this->m_ksubli * this->m_PET[rw];
124  this->m_SE[rw] = min(snow, se);
125  }
126  }
127  }
128  this->m_lastSWE = this->m_swe;
129  return true;
130 }
